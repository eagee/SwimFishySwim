149bool isButtonAvailable(LevelInfo *levelInfo, GameMenuInfo &gameMenu, int ID, int cost)
150{
151 bool returnValue = false;
152
153 if (gameMenu.ButtonList[ID].disabledCounter > 0.0f)
154 {
155 gameMenu.ButtonList[ID].enabled = ENABLED_STATE_JUST_USED;
156 gameMenu.ButtonList[ID].disabledCounter -= (5.0f * gTimeScale);
157 if(gameMenu.ButtonList[ID].disabledCounter < 0.0f)
158 {
159 gameMenu.ButtonList[ID].disabledCounter = 0.0f;
160 }
161 }
162 else if(levelInfo->CurrentEnergyLevel >= cost)
163 {
164 gameMenu.ButtonList[ID].enabled = ENABLED_STATE_ENABLED;
165 returnValue = true;
166 }
167 else if(levelInfo->CurrentEnergyLevel > (cost - 5))
168 {
169 gameMenu.ButtonList[ID].enabled = ENABLED_STATE_ALMOST;
170 }
171 else
172 {
173 gameMenu.ButtonList[ID].enabled = ENABLED_STATE_DISABLED;
174 }
175
176 return returnValue;
177}

182static void clickSpittingTowerButton(LevelInfo *levelInfo)
183{
184 SpriteTower *newTower = new SpriteTower ();
185 Vector location;
186 levelInfo->Sounds[MENU_YES]->playOnce();
187 location = levelInfo->TileMap->getLocation(TILEMAP_LAYER_A);
188 location.x = -location.x;
189 location.y = -location.y;
190 LevelMethodsSetupSprite(levelInfo->Models[SPITTING_FLOWER_TOWER], newTower, ATTACK_TYPE_GUN, location.x, location.y, SPRITE_WATER_LEVEL + 2.1f, 1.0f);
191 newTower->setHealth(100.0f);
192 newTower->setVelocities(0.0f, 0.0f, 0.0f);
193 newTower->assignWeapon(levelInfo->ProjectileWeapons[PROJECTILE_TYPE_SPIT]);
194 levelInfo->TowerSprites.push_back(*newTower);
195 levelInfo->CurrentEnergyLevel -= gBaseSpitTowerCost;
196 levelInfo->PlayerOneSprite->setAnimation(SALUTE);
197 LevelMethodsHandleSplashEffects(levelInfo, levelInfo->PlayerOneSprite, 100);
198 delete newTower;
199}

271void InGameMenu::executeMenu(LevelInfo *levelInfo)
272{
273 bool menuButtonPressed = levelInfo->Controllers->checkButtonPress(0, WII_BUTTON_A);
274
275 // We have to check whether or not the user has enough energy,
276 // and only display the buttons the user has the energy for...
277
278 isButtonAvailable(levelInfo, myGameMenu, CHOMP_TOWER_BUTTON, gBaseChompTowerCost);
279 isButtonAvailable(levelInfo, myGameMenu, DIRECTION_PAD_BUTTON, gBaseChompTowerCost);
280 isButtonAvailable(levelInfo, myGameMenu, THROWING_TOWER_BUTTON, gBaseStunTowerCost);
281 isButtonAvailable(levelInfo, myGameMenu, SPITTING_TOWER_BUTTON, gBaseSpitTowerCost);
282 isButtonAvailable(levelInfo, myGameMenu, WIND_TOWER_BUTTON, gBaseWindTowerCost);
283
284 // Check for the A button to be pressed
285 if (myGameMenu.CurrentButtonState == BUTTON_RELEASED)
286 {
287 myGameMenu.CurrentButtonState = BUTTON_FREE;
288 }
289 else if (myGameMenu.CurrentButtonState == BUTTON_FREE)
290 {
291 myGameMenu.ClickedButtonID = NO_BUTTON_CLICKED;
292 if (menuButtonPressed == true)
293 {
294 myGameMenu.CurrentButtonState = BUTTON_PRESSED;
295 GameMenuMethodsUpdateClickedButtonID(0, &myGameMenu, levelInfo);
296 }
297 }
298 // Once A is pressed, check for it to be released, then once it's released check to see if we were inside
299 // a set of button coordinates...
300 else if ((myGameMenu.CurrentButtonState == BUTTON_PRESSED) && (menuButtonPressed == false))
301 {
302 GameMenuMethodsUpdateClickedButtonID(0, &myGameMenu, levelInfo);
303 switch (myGameMenu.ClickedButtonID)
304 {
305 case SPITTING_TOWER_BUTTON:
306 // Add a new throwing tower to our level (we'll have to find out how many towers we can actually have)
307 if(isButtonAvailable(levelInfo, myGameMenu, myGameMenu.ClickedButtonID, gBaseSpitTowerCost) == true)
308 {
309 clickSpittingTowerButton(levelInfo);
310 //temporarilyDisableAllButtons(myGameMenu);
311 }
312 else
313 {
314 levelInfo->Sounds[MENU_NO]->playOnce();
315 }
316 break;
317 case THROWING_TOWER_BUTTON:
318 // Add a new throwing tower to our level (we'll have to find out how many towers we can actually have)
319 if(isButtonAvailable(levelInfo, myGameMenu, myGameMenu.ClickedButtonID, gBaseStunTowerCost) == true)
320 {
321 clickThrowingTowerButton(levelInfo);
322 //temporarilyDisableAllButtons(myGameMenu);
323 }
324 else
325 {
326 levelInfo->Sounds[MENU_NO]->playOnce();
327 }
328 break;
329 default:
330 break;
331 }
332
333 myGameMenu.CurrentButtonState = BUTTON_RELEASED;
334 }
335
336 GameMethodsRenderButtonsOnly(0, &myGameMenu, levelInfo);
337}
