184void NewTowerMenu::executeMenu(LevelInfo *levelInfo)
185{
186 bool doneLooping = false;
187 CONTROLLER_STATUS status;
188
189 Vector location;
190
191 while(doneLooping == false)
192 {
193 // This is our input code, and should always be run
194 levelInfo->Controllers->updateControllerStates();
195 status = levelInfo->Controllers->checkControllerStatus(0);
196 if(status == CONTROLLER_READY)
197 {
198
199 if(levelInfo->Controllers->checkButtonPress(0, WII_BUTTON_PLUS) == true)
200 {
201#ifdef PLATFORM_IS_WII
202 _break();
203#endif
204 }
205
206 bool menuButtonPressed = levelInfo->Controllers->checkButtonPress(0, WII_BUTTON_A);
207
208 if(levelInfo->Controllers->checkButtonPress(0, WII_BUTTON_B) == true)
209 {
210 // if A and B are pressed at the same time, they want to exit cest menu...
211 levelInfo->Sounds[MENU_YES]->playOnce();
212 doneLooping = true;
213 }
214
215 // Check for the A button to be pressed
216 if(myGameMenu.CurrentButtonState == BUTTON_RELEASED)
217 {
218 myGameMenu.CurrentButtonState = BUTTON_FREE;
219 }
220 else if(myGameMenu.CurrentButtonState == BUTTON_FREE)
221 {
222 myGameMenu.ClickedButtonID = NO_BUTTON_CLICKED;
223 if(menuButtonPressed == true)
224 {
225 myGameMenu.CurrentButtonState = BUTTON_PRESSED;
226 GameMenuMethodsUpdateClickedButtonID(0, &myGameMenu, levelInfo);
227 }
228 }
229 // Once A is pressed, check for it to be released, then once it's released check to see if we were inside
230 // a set of button coordinates...
231 else if((myGameMenu.CurrentButtonState == BUTTON_PRESSED) && (menuButtonPressed == false))
232 {
233 SpriteTower *newTower = new SpriteTower();
234 GameMenuMethodsUpdateClickedButtonID(0, &myGameMenu, levelInfo);
235 switch(myGameMenu.ClickedButtonID)
236 {
237 case FINISHED_BUTTON:
238 levelInfo->Sounds[MENU_YES]->playOnce();
239 doneLooping = true;
240 break;
241 case NEW_SPITTING_TOWER_BUTTON:
242 // Add a new throwing tower to our level (we'll have to find out how many towers we can actually have)
243 levelInfo->Sounds[MENU_YES]->playOnce();
244 doneLooping = true;
245 levelInfo->CurrentUpgradePoints -= gBaseSpitTowerCost;
246 location = levelInfo->TileMap->getLocation(TILEMAP_LAYER_A);
247 location.x =-location.x; location.y =-location.y;
248 LevelMethodsSetupSprite(levelInfo->Models[PRIMARY_FISH_MODEL], newTower, ATTACK_TYPE_GUN, location.x, location.y, SPRITE_WATER_LEVEL, 1.0f);
249 newTower->setHealth(100.0f);
250 newTower->setVelocities(0.0f, 0.0f, 0.0f);
251 newTower->assignWeapon(levelInfo->ProjectileWeapons[PROJECTILE_TYPE_SPIT]);
252 levelInfo->TowerSprites.push_back(*newTower);
253 break;
254 case NEW_THROWING_TOWER_BUTTON:
255 // Add a new throwing tower to our level (we'll have to find out how many towers we can actually have)
256 levelInfo->Sounds[MENU_YES]->playOnce();
257 doneLooping = true;
258 levelInfo->CurrentUpgradePoints -= gBaseStunTowerCost;
259 location = levelInfo->TileMap->getLocation(TILEMAP_LAYER_A);
260 location.x = -location.x;
261 location.y = -location.y;
262 LevelMethodsSetupSprite(levelInfo->Models[ATTACKING_BADDIE_FISH_MODEL], newTower, ATTACK_TYPE_GUN, location.x, location.y, SPRITE_WATER_LEVEL, 1.0f);
263 newTower->setHealth(100.0f);
264 newTower->setVelocities(0.0f, 0.0f, 0.0f);
265 newTower->assignWeapon(levelInfo->ProjectileWeapons[PROJECTILE_TYPE_THROW]);
266 levelInfo->TowerSprites.push_back(*newTower);
267 break;
268 case ENERGIZE_TOWERS_BUTTON:
269 // Energize all of the existing towers
270 levelInfo->Sounds[MENU_YES]->playOnce();
271 doneLooping = true;
272 levelInfo->ProjectileWeapons[PROJECTILE_TYPE_SPIT]->modifyWeaponExperienceBy(500);
273 levelInfo->ProjectileWeapons[PROJECTILE_TYPE_THROW]->modifyWeaponExperienceBy(500);
274 levelInfo->CurrentUpgradePoints -= gEnergizeTowersCost;
275 break;
276
277 default:
278 break;
279 }
280 delete newTower;
281
282 myGameMenu.CurrentButtonState = BUTTON_RELEASED;
283 }
284
285 GameMenuMethodsRenderMenu(0, &myGameMenu, levelInfo);
286 }
287 //TODO: We need a dialog that informs the user that the controller is not responding...
288 }
