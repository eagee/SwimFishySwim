39void profileStart(std::string methodName)
40{
41 gMethodIterator = gMethodData.find(methodName);
42
43 // If we found the method already, then update the starting time...
44 // otherwise add a new element into the map based on the method name...
45 if(gMethodIterator != gMethodData.end())
46 {
47 ASSERT(gMethodData[methodName].StartedAtMs == 0);
48#ifdef PLATFORM_IS_WII
49 gMethodData[methodName].StartedAtMs = (gettime() / TB_TIMER_CLOCK);
50#endif
51#ifdef PLATFORM_IS_WIN
52 gMethodData[methodName].StartedAtMs = (float)GetTickCount();
53#endif
54 }
55 else
56 {
57 MethodProfile newMethod;
58 newMethod.MethodName = methodName;
59
60#ifdef PLATFORM_IS_WII
61 newMethod.StartedAtMs = (gettime() / TB_TIMER_CLOCK);
62#endif
63#ifdef PLATFORM_IS_WIN
64 gMethodData[methodName].StartedAtMs = (float)GetTickCount();
65#endif
66 newMethod.TotalMsElapsed = 0;
67 gMethodData[methodName] = newMethod;
68 }
69}

74void profileStop(std::string methodName)

76 gMethodIterator = gMethodData.find(methodName);
77
78 // If we found the method already, then update the starting time...
79 // otherwise add a new element into the map based on the method name...
80 if (gMethodIterator != gMethodData.end())
81 {
82#ifdef PLATFORM_IS_WII
83 float elapsedThisRun = ((gettime() / TB_TIMER_CLOCK) - gMethodData[methodName].StartedAtMs);
84#endif
85#ifdef PLATFORM_IS_WIN
86 float elapsedThisRun = GetTickCount() - gMethodData[methodName].StartedAtMs;
87#endif
88 gMethodData[methodName].TotalMsElapsed += elapsedThisRun;
89 gMethodData[methodName].StartedAtMs = 0;
