505void LevelOne::releaseBaddiesOnASchedule(float counterIncrement)
506{
507 //printf("\x1b[2;0H"); printf("Wave: %d, Idx: %d Max: %d", myLevelInfo->BaddieDetails.WaveIndex, myLevelInfo->BaddieDetails.ReleaseIndex, myLevelInfo->BaddieDetails.MaximumThisWave);
508 myLevelInfo->BaddieDetails.ReleaseDelay += counterIncrement;
509
510 // Go through our baddies. They're all the same on this level...
511 int maxBaddies = MAX_BADDIES_AT_A_TIME;
512 if(myLevelInfo->BaddieDetails.WaveIndex == 2) maxBaddies = 12;
513 if((myLevelInfo->BaddieDetails.ReleaseDelay >= 1.0f) && (myLevelInfo->BaddieDetails.CurrentlyAlive < maxBaddies))
514 {
515 handleReleaseDelay(myLevelInfo, 9.0f);
516
517 // If we've already released all our baddies, check to see if they're all dead,
518 // and increment to the next available wave if they are :D :D
519 if(myLevelInfo->BaddieDetails.ReleaseIndex >= myLevelInfo->BaddieDetails.MaximumThisWave)
520 {
521 if(myLevelInfo->BaddieDetails.BaddiesThisWaveAreDead == true)
522 {
523 handleEndOfLevel(myLevelInfo);
524 }
525 }
526
527 if(myLevelInfo->BaddieDetails.ReleaseIndex < myLevelInfo->BaddieDetails.MaximumThisWave)
528 {
529 myLevelInfo->BaddieDetails.ReleaseIndex++;
530 if(myLevelInfo->BaddieDetails.WaveIndex == 1)
531 {
532 if(myLevelInfo->BaddieDetails.ReleaseIndex <= 6)
533 handleBaddieRelease(myLevelInfo, SMALL_BADDIE_FISH_MODEL, 0.20f, LEVEL_ONE_BADDIE_VALUE, ATTACK_TYPE_NONE, myLevelOneSpawnPoint, L1_RIGHT);
534 else if(myLevelInfo->BaddieDetails.ReleaseIndex <= 12)
535 handleBaddieRelease(myLevelInfo, SMALL_BADDIE_FISH_MODEL, 0.20f, LEVEL_ONE_BADDIE_VALUE, ATTACK_TYPE_NONE, myLevelOneSpawnPoint, L1_LEFT);
536 else if(myLevelInfo->BaddieDetails.ReleaseIndex <= 18)
537 handleBaddieRelease(myLevelInfo, SMALL_BADDIE_FISH_MODEL, 0.20f, LEVEL_ONE_BADDIE_VALUE, ATTACK_TYPE_NONE, myLevelOneSpawnPoint, L1_CENTER);
538 else if(myLevelInfo->BaddieDetails.ReleaseIndex <= 24)
539 handleBaddieRelease(myLevelInfo, SMALL_BADDIE_FISH_MODEL, 0.20f, LEVEL_ONE_BADDIE_VALUE, ATTACK_TYPE_NONE, myLevelOneSpawnPoint, L1_RANDOM);
540 }
541 else if(myLevelInfo->BaddieDetails.WaveIndex == 2)

697 // We need to trigger a single update to make sure our baddie renders...

701 // Setup our baddie for rendering during an earthquake

729 if((interpolation >= 0.5f) && (interpolation <= 0.51f))
730 {
731 myLevelInfo->TileMap->shakeMapFor(3, 0.20f);
732 myLevelInfo->Sounds[PICKLE_DAMAGE]->playOnce();
733 LevelMethodsHandleWarpEffects(myLevelInfo, &myLevelInfo->BaddieDetails.Sprites.at(myLevelInfo->BaddieDetails.ReleaseIndex));

738
739 myLevelInfo->Sounds[PICKLE_DAMAGE]->playOnce();

741 LevelMethodsHandleWarpEffects(myLevelInfo, &myLevelInfo->BaddieDetails.Sprites.at(myLevelInfo->BaddieDetails.ReleaseIndex));
742 myLevelInfo->TileMap->shakeMapFor(12, 0.25);
743
744 // Render the baddie rising from the depths after the earthquake...
745 for(float interpolation = 0.0f; interpolation <= 1.0f; interpolation += 0.0025f)

747 if(baddieLocation.z > myLevelOneSpawnPoint.z)
748 {
749 baddieLocation.z -= (0.04f);

759 myLevelInfo->Sounds[PICKLE_DAMAGE]->playOnce();

763 LevelMethodsHandleSplashEffects(myLevelInfo, &myLevelInfo->BaddieDetails.Sprites.at(myLevelInfo->BaddieDetails.ReleaseIndex), 10);
764 renderCameraSweep(initialMapLocation, interpolatedALocation, interpolatedDLocation);
765 }
766
767 // Swing the camera back over to our good guy ...
768 for(float interpolation = 1.0f; interpolation >= 0.0f; interpolation -= 0.008f)
769 {
770 targetMapLocation.z = lastAMapLocation.z;
771 fastVectorSubtract(&targetMapLocation, &lastAMapLocation, &interpolatedALocation);

779
780 myLevelInfo->SpriteOffset = myLevelInfo->TileMap->getLocation(TILEMAP_LAYER_A);
781 myLevelInfo->SpriteOffset.z = 0.0f;
782 myLevelInfo->BaddieDetails.Sprites.at(myLevelInfo->BaddieDetails.ReleaseIndex).setOffset(myLevelInfo->SpriteOffset);
783 myLevelInfo->Graphics->setCameraTarget(myLevelInfo->PlayerOneSprite->getLocation(), myLevelInfo->PlayerOneSprite->getAngle(), 30.0f);
784
785 renderCameraSweep(initialMapLocation, interpolatedALocation, interpolatedDLocation);
786 }
787
788 myLevelInfo->PlayerOneSprite->setOffset(VEC_INIT(0.0f, 0.0f, 0.0f));

790 // Reset our map to it's original location
791 myLevelInfo->TileMap->translateMap(TILEMAP_LAYER_A, lastAMapLocation);

793 myLevelInfo->SpriteOffset = myLevelInfo->TileMap->getLocation(TILEMAP_LAYER_A);
794 myLevelInfo->SpriteOffset.z = 0.0f;
