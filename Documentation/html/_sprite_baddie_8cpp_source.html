364void SpriteBaddie::modifyHealthBy(float healthModifier)
365{
366 mySpriteInfo.Health += healthModifier;
367 if(mySpriteInfo.Health > mySpriteInfo.MaxHealth) mySpriteInfo.Health = mySpriteInfo.MaxHealth;
368 // Pain A moves into death in the health <= 0 inside of the spriteMethods animation codeb
369 if((mySpriteInfo.CurrentAnimation != DEATH_FALLBACKSLOW)
370 && (mySpriteInfo.CurrentAnimation != DEATH_FALLBACK)
371 && (mySpriteInfo.CurrentAnimation != DEATH_FALLFORWARD)
372 && (mySpriteInfo.CurrentAnimation != PAIN_A)
373 && (mySpriteInfo.CurrentAnimation != PAIN_B)
374 && (mySpriteInfo.CurrentAnimation != PAIN_C))
375 {
376 int painAnim = rand() % 2;
377 if(painAnim == 0)
378 {
379 mySpriteInfo.CurrentAnimation = PAIN_A;
380 }
381 else if(painAnim == 1)
382 {
383 mySpriteInfo.CurrentAnimation = PAIN_B;
384 }
385 else if(painAnim == 2)
386 {
387 mySpriteInfo.CurrentAnimation = PAIN_C;
388 }
389 }
390}
