138void moveCoordsByPitchAngle(float &x, float &y, float pitch)
139{
140 float speed = 3;
141
142 x = x + speed * (float)cos(pitch);
143 y = y + speed * (float)sin(pitch);
144
145 if(x > 80) x = 0;
146 if(y > 60) y = 0;
147 if(x < 0) x = 80;
148 if(y < 0) y = 60;
149}
