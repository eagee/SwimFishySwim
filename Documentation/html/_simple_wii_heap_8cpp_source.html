47void *simpleWiiHeapAllocatePersistentMemory(size_t memorySize)
48{
49 unsigned long returnValue = gMemoryAvailable.startAddress;

51 if((returnValue + memorySize) >= gMemoryAvailable.endAddress)

53 ASSERT(false);
54 throw std::bad_alloc();
55 }
56
57 gMemoryAvailable.startAddress += memorySize;
58
59 // Figure out how many bytes we need to pad the memory by to keep it aligned...
