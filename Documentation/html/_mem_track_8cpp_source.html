121 /* ---------------------------------------- BlockHeader AddNode */
122
123 void BlockHeader::AddNode(BlockHeader *node)
124 {
125 ASSERT(node != NULL);
126 ASSERT(node->myPrevNode == NULL);
127 ASSERT(node->myNextNode == NULL);
128
129 // If we have at least one node in the list ...
130 if (ourFirstNode != NULL)
131 {
132 // ... make the new node the first node's predecessor.
133 ASSERT(ourFirstNode->myPrevNode == NULL);
134 ourFirstNode->myPrevNode = node;
135 }
136
137 // Make the first node the new node's succesor.
138 node->myNextNode = ourFirstNode;
139
140 // Make the new node the first node.
141 ourFirstNode = node;
142 }
143
144 /* ---------------------------------------- BlockHeader RemoveNode */
145
146 void BlockHeader::RemoveNode(BlockHeader *node)
147 {
148 ASSERT(node != NULL);
149 ASSERT(ourFirstNode != NULL);
150
151 // If target node is the first node in the list...
152 if (ourFirstNode == node)
153 {
154 // ... make the target node's successor the first node.
155 ASSERT(ourFirstNode->myPrevNode == NULL);
156 ourFirstNode = node->myNextNode;
157 }
158
159 // Link target node's predecessor, if any, to its successor.
160 if (node->myPrevNode != NULL)
161 {
162 node->myPrevNode->myNextNode = node->myNextNode;
163 }
164
165 // Link target node's successor, if any, to its predecessor.
166 if (node->myNextNode != NULL)
167 {
168 node->myNextNode->myPrevNode = node->myPrevNode;
169 }
170
171 // Clear target node's previous and next pointers.
172 node->myPrevNode = NULL;
173 node->myNextNode = NULL;
174 }

214 class Signature
215 {
216 private: // constants
217 static const unsigned int SIGNATURE1 = 0xCAFEBABE;
218 static const unsigned int SIGNATURE2 = 0xFACEFACE;
219
220 private: // member variables
221 unsigned int mySignature1;
222 unsigned int mySignature2;
223
224 public: // construction/destruction
225 Signature() : mySignature1(SIGNATURE1), mySignature2(SIGNATURE2) {};
226 ~Signature() { mySignature1 = 0; mySignature2 = 0; }
227
228 public: // static member functions
229 static bool IsValidSignature(const Signature *pProspectiveSignature)
230 {
231 try
232 {
233 if (pProspectiveSignature->mySignature1 != SIGNATURE1) return false;
234 if (pProspectiveSignature->mySignature2 != SIGNATURE2) return false;
235 return true;
236 }
237 catch (...)
238 {
239 return false;
240 }
241 }

244 /* ------------------------------------------------------------ */
245 /* -------------------- address conversion -------------------- */
246 /* ------------------------------------------------------------ */
247
248 /* We divide the memory blocks we allocate into two "chunks", the
249 * "prolog chunk" where we store information about the allocation,
250 * and the "user chunk" which we return to the caller to use.
251 */
252
253 /* ---------------------------------------- alignment */
254
255 const size_t ALIGNMENT = 4;
256
257 /* If "value" (a memory size or offset) falls on an alignment boundary,
258 * then just return it. Otherwise return the smallest number larger
259 * than "value" that falls on an alignment boundary.
260 */
261
262 #define PAD_TO_ALIGNMENT_BOUNDARY(value) \
263 ((value) + ((ALIGNMENT - ((value) % ALIGNMENT)) % ALIGNMENT))
264
265 /* ---------------------------------------- chunk structs */
266
267 /* We declare incomplete structures for each chunk, just to
268 * provide type safety.
269 */
270
271 struct PrologChunk;
272 struct UserChunk;
273
274 /* ---------------------------------------- chunk sizes and offsets */
275
276 const size_t SIZE_BlockHeader = PAD_TO_ALIGNMENT_BOUNDARY(sizeof(BlockHeader));
277 const size_t SIZE_Signature = PAD_TO_ALIGNMENT_BOUNDARY(sizeof(Signature));
278
279 const size_t OFFSET_BlockHeader = 0;
280 const size_t OFFSET_Signature = OFFSET_BlockHeader + SIZE_BlockHeader;
281 const size_t OFFSET_UserChunk = OFFSET_Signature + SIZE_Signature;
282
283 const size_t SIZE_PrologChunk = OFFSET_UserChunk;
284
285 /* ---------------------------------------- GetUserAddress */
286
287 static UserChunk *GetUserAddress(PrologChunk *pProlog)
288 {
289 char *pchProlog = reinterpret_cast<char *>(pProlog);
290 char *pchUser = pchProlog + OFFSET_UserChunk;
291 UserChunk *pUser = reinterpret_cast<UserChunk *>(pchUser);
292 return pUser;
293 }
294
295 /* ---------------------------------------- GetPrologAddress */
296
297 static PrologChunk *GetPrologAddress(UserChunk *pUser)
298 {
299 char *pchUser = reinterpret_cast<char *>(pUser);
300 char *pchProlog = pchUser - OFFSET_UserChunk;
301 PrologChunk *pProlog = reinterpret_cast<PrologChunk *>(pchProlog);
302 return pProlog;
303 }
304
305 /* ---------------------------------------- GetHeaderAddress */
306
307 static BlockHeader *GetHeaderAddress(PrologChunk *pProlog)
308 {
309 char *pchProlog = reinterpret_cast<char *>(pProlog);
310 char *pchHeader = pchProlog + OFFSET_BlockHeader;
311 BlockHeader *pHeader = reinterpret_cast<BlockHeader *>(pchHeader);
312 return pHeader;
313 }
314
315 /* ---------------------------------------- GetSignatureAddress */
316
317 static Signature *GetSignatureAddress(PrologChunk *pProlog)
318 {
319 char *pchProlog = reinterpret_cast<char *>(pProlog);
320 char *pchSignature = pchProlog + OFFSET_Signature;
321 Signature *pSignature = reinterpret_cast<Signature *>(pchSignature);
322 return pSignature;
323 }

329 /* ---------------------------------------- TrackMalloc */
330
331 void *TrackMalloc(size_t size)
332 {
333 // Allocate the memory, including space for the prolog.
334 PrologChunk *pProlog = (PrologChunk *)malloc(SIZE_PrologChunk + size);
335
336 // If the allocation failed, then return NULL.
337 if (pProlog == NULL) return NULL;
338
339 // Use placement new to construct the block header in place.
340 BlockHeader *pBlockHeader = new (pProlog) BlockHeader(size);
341
342 // Link the block header into the list of extant block headers.
343 BlockHeader::AddNode(pBlockHeader);
344
345 // Use placement new to construct the signature in place.
346 Signature *pSignature = new (GetSignatureAddress(pProlog)) Signature;
347
348 // Get the offset to the user chunk and return it.
349 UserChunk *pUser = GetUserAddress(pProlog);
350
351 return pUser;
352 }

354 /* ---------------------------------------- TrackFree */
355
356 void TrackFree(void *p)
357 {
358 // It's perfectly valid for "p" to be null; return if it is.
359 if (p == NULL) return;
360
361 // Get the prolog address for this memory block.
362 UserChunk *pUser = reinterpret_cast<UserChunk *>(p);
363 PrologChunk *pProlog = GetPrologAddress(pUser);
364
365 // Check the signature, and if it's invalid, return immediately.
366 Signature *pSignature = GetSignatureAddress(pProlog);
367 if (!Signature::IsValidSignature(pSignature)) return;
368
369 // Destroy the signature.
370 pSignature->~Signature();
371 pSignature = NULL;
372
373 // Unlink the block header from the list and destroy it.
374 BlockHeader *pBlockHeader = GetHeaderAddress(pProlog);
375 BlockHeader::RemoveNode(pBlockHeader);
376 pBlockHeader->~BlockHeader();
377 pBlockHeader = NULL;
378
379 // Free the memory block.
380 free(pProlog);
381 }

383 /* ---------------------------------------- TrackStamp */
384
385 void TrackStamp(void *p, const MemStamp &stamp, char const *typeName)
386 {
387 // Get the header and signature address for this pointer.
388 UserChunk *pUser = reinterpret_cast<UserChunk *>(p);
389 PrologChunk *pProlog = GetPrologAddress(pUser);
390 BlockHeader *pHeader = GetHeaderAddress(pProlog);
391 Signature *pSignature = GetSignatureAddress(pProlog);
392
393 // If the signature is not valid, then return immediately.
394 if (!Signature::IsValidSignature(pSignature)) return;
395
396 // "Stamp" the information onto the header.
397 pHeader->Stamp(stamp.filename, stamp.lineNum, typeName);
398 }

400 /* ---------------------------------------- TrackDumpBlocks */
401
402 void TrackDumpBlocks()
403 {
404 // Get an array of pointers to all extant blocks.
405 size_t numBlocks = BlockHeader::CountBlocks();
406 BlockHeader **ppBlockHeader =
407 (BlockHeader **)calloc(numBlocks, sizeof(*ppBlockHeader));

409

418 BlockHeader *pBlockHeader = ppBlockHeader[i];

420 size_t size = pBlockHeader->GetRequestedSize();
421 char const *fileName = pBlockHeader->GetFilename();

424 trace("... %s:%d\n", fileName, lineNum);

444 /* ---------------------------------------- SummarizeMemoryUsageForType */
445
446 static void SummarizeMemoryUsageForType(
447 MemDigest *pMemDigest,
448 BlockHeader **ppBlockHeader,
449 size_t startPost,
450 size_t endPost
451 )
452 {
453 pMemDigest->typeName = ppBlockHeader[startPost]->GetTypeName();
454 pMemDigest->blockCount = 0;
455 pMemDigest->totalSize = 0;
456 for (size_t i = startPost; i < endPost; i++)
457 {
458 pMemDigest->blockCount++;
459 pMemDigest->totalSize += ppBlockHeader[i]->GetRequestedSize();
460 ASSERT(strcmp(ppBlockHeader[i]->GetTypeName(), pMemDigest->typeName) == 0);
461 }

469 size_t numBlocks = BlockHeader::CountBlocks();

484 // Find out how many unique types we have.

488 char const *prevTypeName = ppBlockHeader[i - 1]->GetTypeName();
