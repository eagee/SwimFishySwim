48void ProjectileMethodsActivate(int weaponLevel, float xyModifier, float angleOffset, ISprite *projectile, ISprite *triggeredBySprite, Vector &spriteOffset, IAudio *soundToPlay)
49{
50 Vector newLocation = triggeredBySprite->getLocation();
51 DIRECTION newDirection = LevelMethodsGetAngleDirection(triggeredBySprite->getControlAngle(), 1.0f);
52
53 // Increase our location to xymodifier in front of the direction the sprite is facing...
54 // as well as change our offset depending on our weapon level number...
55 float angleDegrees = modifyAngleByDegrees(triggeredBySprite->getAngle(), angleOffset);
56 float angleRads = getRadiansForAngle(angleDegrees);
57 float moveXBy = -(xyModifier * cos(angleRads));
58 float moveYBy = -(xyModifier * sin(angleRads));
59
60 newLocation.x -= spriteOffset.x;
61 newLocation.y -= spriteOffset.y;
62
63
64 // bomb: newLocation.z -= 0.7f;
65 // shot: newLocation.z -= 0.6f;
66
67 newLocation.x += moveXBy;
68 newLocation.y += moveYBy;
69 float waterLevel = triggeredBySprite->getWaterLevel();
70 newLocation.z = waterLevel + spriteOffset.z;
71
72 angleDegrees = modifyAngleByDegrees(triggeredBySprite->getControlAngle(), angleOffset);
73 projectile->setDirection(newDirection);
74 projectile->translateSprite(newLocation);
75 projectile->setWaterLevel(20.8f);
76 projectile->setOffset(spriteOffset);
77 projectile->setAngle(angleDegrees);
78 projectile->setJump(JUMP_STATE_NOT_READY);
79 projectile->setJump(JUMP_STATE_READY);
80 soundToPlay->playOnce();
81}

149void ProjectileMethodsUpdateProjectiles(LevelInfo* levelInfo, ProjectileWeaponInfo &weaponInfo)
150{
151 std::vector<SpriteBullet>::iterator spriteIterator = weaponInfo.Projectiles.begin();
152 for(int idx = 0; idx < weaponInfo.CurrentMaxProjectiles; idx++)
153 {
154 // Handle all the controls and inputs for our baddie sprites!
155 if(spriteIterator->isDead() == false)
156 {
157 JUMP_STATE lastJumpState = spriteIterator->getJump();
158 spriteIterator->setOffset(levelInfo->SpriteOffset);
159 spriteIterator->increaseAcceleration();
160 spriteIterator->setControlMagnitude(2.0f);
161 LevelMethodsHandleTilemapCollision(*spriteIterator, levelInfo);
162 LevelMethodsHandleRippleEffects(levelInfo, &*spriteIterator);
163 spriteIterator->updateLocationByVelocity();
164 if ((lastJumpState == JUMP_STATE_LANDING) && (spriteIterator->getJump() != JUMP_STATE_LANDING))
165 {
166 LevelMethodsHandleSplashEffects(levelInfo, &*spriteIterator, 3);
167 }
168 }
169 spriteIterator++;
170 }
171}

186PROJECTILE_LEVEL_STATE ProjectileMethodsUpdateExperience(ProjectileWeaponInfo &weaponInfo, int totalLevels, int maximumProjectiles)
187{
188 PROJECTILE_LEVEL_STATE returnValue = PROJECTILE_LEVEL_SAME;
189 float targetPercent = ((weaponInfo.WeaponExperiencePoints / weaponInfo.MaxWeaponExperiencePoints) * 100.0f);
190 //float lastPercentageComplete = weaponInfo.PercentageComplete;
191 //int lastWeaponLevel = weaponInfo.WeaponLevelNumber;
192
193 // Adjust our weapon percentage incrementally each update (for display purposes)
194 if(weaponInfo.PercentageComplete < targetPercent - 1.0f)
195 {
196 weaponInfo.PercentageComplete += 10.0f * gTimeScale;
197 }
198 else if(weaponInfo.PercentageComplete > targetPercent + 1.0f)
199 {
200 weaponInfo.PercentageComplete -= 10.0f * gTimeScale;
201 }
202 else
203 {
204 weaponInfo.PercentageComplete = targetPercent;
205 }
206
207 if(weaponInfo.PercentageComplete <= 25.0f)
208 {
209 weaponInfo.WeaponLevelNumber = 1;
210 ProjectileMethodsCalculateMaxProjectiles(weaponInfo, totalLevels, maximumProjectiles);
211 }
212 else if(weaponInfo.PercentageComplete <= 50.0f)
213 {
214 weaponInfo.WeaponLevelNumber = 2;
215 ProjectileMethodsCalculateMaxProjectiles(weaponInfo, totalLevels, maximumProjectiles);
216 }
217 else if(weaponInfo.PercentageComplete <= 75.0f)
218 {
219 weaponInfo.WeaponLevelNumber = 3;
220 ProjectileMethodsCalculateMaxProjectiles(weaponInfo, totalLevels, maximumProjectiles);
221 }
222 else if(weaponInfo.PercentageComplete > 75.0f)
223 {
224 weaponInfo.WeaponLevelNumber = 4;
225 ProjectileMethodsCalculateMaxProjectiles(weaponInfo, totalLevels, maximumProjectiles);
226 }
227
228 return returnValue;
229}

250void ProjectileMethodsHandleIfSpriteDead(ISprite &sprite, ProjectileWeaponInfo &weaponInfo)
251{
252 // If the player killed this sprite, drop a power up
253 // If a tower killed this sprite, then add experience to that tower
254 if ((sprite.getHealth() <= 0.0f) && (sprite.getHealth() >= -100.0f))
255 {
256
257 if (weaponInfo.triggeredByType == TOWER_SPRITE)
258 {
259 // Only increase experience randomly, so that it has a chance to go down
260 // and provides a random element :D
261 if(rand() % 3 != 0)
262 {
263 if(weaponInfo.WeaponExperiencePoints < weaponInfo.MaxWeaponExperiencePoints)
264 {
265 weaponInfo.WeaponExperiencePoints += 7.0f;
266 }
267 }
268 }
269 else
270 {
271 // This is the kind of power up created by our gun...
272 SPRITE_POWER_UP powerUpType = (SPRITE_POWER_UP) (rand() % POWER_UP_GUN_LARGE);
273 sprite.setPowerUpType(powerUpType);
274 }
275 sprite.setHealth(-200);
276
277 }
