178void preAllocateSpriteObjects(LevelInfo *levelInfo)
179{
180 // Preallocate Sprites that will remain consistent between levels...
181
182 // Setup the projectile weapons that will persist throughout the game ...
183 levelInfo->ProjectileWeapons[PROJECTILE_TYPE_SHOT] = new ProjectileShot(levelInfo->Models[SHOT_MODEL], levelInfo->Sounds[LEVEL_UP], levelInfo->Sounds[MENU_NO], levelInfo->Sounds[BWOOP]);
184 levelInfo->ProjectileWeapons[PROJECTILE_TYPE_BOMB] = new ProjectileBomb(levelInfo->Models[BOMB_MODEL], levelInfo->Sounds[LEVEL_UP], levelInfo->Sounds[MENU_NO], levelInfo->Sounds[BWOOP]);
185 levelInfo->ProjectileWeapons[PROJECTILE_TYPE_MINE] = new ProjectileMine(levelInfo->Models[MINE_MODEL], levelInfo->Sounds[LEVEL_UP], levelInfo->Sounds[MENU_NO], levelInfo->Sounds[BWOOP]);
186 levelInfo->ProjectileWeapons[PROJECTILE_TYPE_SPIT] = new ProjectileSpit(levelInfo->Models[SHOT_MODEL], levelInfo->Sounds[LEVEL_UP], levelInfo->Sounds[MENU_NO], levelInfo->Sounds[BWOOP]);
187 levelInfo->ProjectileWeapons[PROJECTILE_TYPE_THROW] = new ProjectileToss(levelInfo->Models[TOSS_MODEL], levelInfo->Sounds[LEVEL_UP], levelInfo->Sounds[MENU_NO], levelInfo->Sounds[BWOOP]);
188
189 // Setup our Player Sprite
190 levelInfo->PlayerOneSprite = new SpritePlayer();
191 LevelMethodsSetupSprite(levelInfo->Models[PRIMARY_FISH_MODEL], levelInfo->PlayerOneSprite, ATTACK_TYPE_BOTH, 0.0f, 0.0f, SPRITE_WATER_LEVEL, 1.6f);
192
193 // Setup our Pickle Sprite
194 levelInfo->GreatPickle = new SpriteTarget();
195 LevelMethodsSetupSprite(levelInfo->Models[PICKLE_MODEL], levelInfo->GreatPickle, ATTACK_TYPE_NONE, 36.0f, 30.0f, 19.6f, 1.0f);
196 levelInfo->GreatPickle->setHealth(100.0f);
197 levelInfo->GreatPickle->setVelocities(0.0f, 0.0f, 0.0f);
198
199 levelInfo->CurrentProjectileWeapon = PROJECTILE_TYPE_SHOT;
200 levelInfo->ProjectileWeapon = levelInfo->ProjectileWeapons[levelInfo->CurrentProjectileWeapon];
201
202 // Preallocate powerup sprites (also consistent between levels...)
203 for(int powerupIndex = 0; powerupIndex < MAXIMUM_POWERUPS; powerupIndex++)
204 {
205 SpritePowerup newSprite;
206 levelInfo->PowerUps.PowerUpSprites.push_back(newSprite);
207 LevelMethodsSetupSprite(levelInfo->Models[POWERUP_MODEL], &levelInfo->PowerUps.PowerUpSprites.at(powerupIndex), ATTACK_TYPE_NONE, 27.0f, 20.0f, SPRITE_WATER_LEVEL, 1.0f);
208 }
209}

341void RunGameMain()
342{
343 // Set up our essential system objects that will be used by all of our ILevel implementations...
344#ifdef PLATFORM_IS_WII
345 IGraphics *graphics = new GxGraphics();
346 ControllerWii controllers(720,480);
347#endif
348#ifdef PLATFORM_IS_WIN
349 IGraphics *graphics;// = new GxGraphics();
350 ControllerWin controllers(720,480);
351#endif
352
353 gTextDisplay = new Text(graphics);
354 setupTextRendering(*gTextDisplay, *graphics);
355
356 // Set up a base level info object that is passed to each ILevel implementation...
357 LevelInfo *levelInfo = new LevelInfo();
358 levelInfo->Graphics = graphics;
359 levelInfo->TextOutput = gTextDisplay;
360 levelInfo->Controllers = &controllers;
361
362 printf("\x1b[2;0H");printf("\n\nWaiting for GDB Connection...");
363 //_break();
364
365 // At this point we can load and display our title screen / start the title music...
366 //LevelMethodsPreAllocateLevelMusic(MP3_AUDIO, levelInfo, "//assets//music//fishyPop.mp3");
367 LevelMethodsPrepareSplashScreen(levelInfo, 100.0f);
368
369 // Now that all our data is loaded we can just wait for a keypress...
370 //TODO: WAIT HERE FOR A KEY TO BE PRESSED
371 //while(levelInfo->Controllers->checkButtonPress(0, WII_BUTTON_A) != true)
372 //{
373 // levelInfo->Controllers->updateControllerStates();
374 // LevelMethodsDisplaySplashScreen(levelInfo);
375 //}
376
377 LevelMethodsPrepareSplashScreen(levelInfo, 4.0f);
378
379 // Preallocate all of the sprites, models, textures, music, and sound that will persist throughout the game ...
380 preAllocateAudioData(levelInfo);
381 preAllocateGameModels(levelInfo);
382 preAllocateSpriteObjects(levelInfo);
383 preAllocateMenuData(levelInfo);
384 preAllocateParticleData(levelInfo);
385
386 // Free up the resources our title screen was using, and load the in game mp3 data..
387 LevelMethodsDeAllocateSplashScreen(levelInfo);
388
389 // Load our tilemap and then play level one!
390 LevelOne levelOne(*levelInfo);
391 levelOne.runUntilLevelIsFinished();
392
393 // TODO: Deallocate our tilemap, load another one, and play level two!
394
395 // Deallocate all of the sprites, models, textures, music, and sound that persisted throughout the game ...
396 deAllocateSpriteObjects(levelInfo);
397 deAllocateGameModels(levelInfo);
398 deAllocateAudioData(levelInfo);
399 deAllocateMenuData(levelInfo);
400 deAllocateParticleData(levelInfo);
401
402 delete gTextDisplay;
403}
