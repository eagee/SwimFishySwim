130Vector SpritePowerup::updateLocationByVelocity()
131{
132 // Reduce our health as we move ...
133 mySpriteInfo.Health -= 0.01f;
134
135 if(mySpriteInfo.Health <= 0.0f)
136 {
137 mySpriteInfo.CurrentAnimation = DEATH_FALLBACKSLOW;
138 if(myPowerupSize > 0)
139 {
140 myPowerupSize -= (2.0f * gTimeScale);
141 }
142 if(myPowerupSize <= 0) mySpriteInfo.Dead = true;
143 Vector zeroVector = {0.0f, 0.0f, 0.0f};
144 return zeroVector;
145 }
146 else
147 {
148 return spriteMethodsUpdateLocationByVelocity(&mySpriteInfo);
149 }

309void SpritePowerup::renderSprite(IGraphics *graphics)
310{
311
312 if (myPowerupSize <= 1.0f)
313 {
314 myPowerupSize += 1.0f * gTimeScale;
315 // Limit the power up size based on the level of the sprite ...
316 if (mySpriteInfo.SpriteLevel <= 0)
317 {
318 if (myPowerupSize > 0.5f)
319 myPowerupSize = 0.5f;
320 }
321 else if (mySpriteInfo.SpriteLevel == 1)
322 {
323 if (myPowerupSize > 0.75f)
324 myPowerupSize = 0.75f;
325 }
326 else if (mySpriteInfo.SpriteLevel == 2)
327 {
328 if (myPowerupSize > 1.0f)
329 myPowerupSize = 1.0f;
330 }
331 else if (mySpriteInfo.SpriteLevel >= 3)
332 {
333 if (myPowerupSize > 10.0f)
334 myPowerupSize = 10.0f;
335 }
336
337 mySpriteInfo.BoundingTriangleSize = (myPowerupSize / 2);
338 }
339
340 //myRotationValue += 4.0f;
341 //Rotate rotateBy = {{0.0, 0.0, 1.0}, myRotationValue};
342 //graphics->rotateNextObjectInScene(&rotateBy);
343
344 spriteMethodsProcessAnimation(&mySpriteInfo);
345 spriteMethodsRenderSprite(&mySpriteInfo, graphics, myOffsetValue, false);
346 //spriteMethodsRenderBulletSprite(&mySpriteInfo, graphics, myOffsetValue);
347 mySpriteInfo.ControlMagnitude = 1.0f;
348
349}
