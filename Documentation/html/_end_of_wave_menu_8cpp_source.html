151void EndOfWaveMenu::executeMenu(LevelInfo *levelInfo)
152{
153 bool doneLooping = false;
154 CONTROLLER_STATUS status;
155
156 while(doneLooping == false)
157 {
158 // This is our input code, and should always be run
159 levelInfo->Controllers->updateControllerStates();
160 status = levelInfo->Controllers->checkControllerStatus(0);
161 if(status == CONTROLLER_READY)
162 {
163
164 if(levelInfo->Controllers->checkButtonPress(0, WII_BUTTON_PLUS) == true)
165 {
166#ifdef PLATFORM_IS_WII
167 _break();
168#endif
169 }
170
171
172 bool menuButtonPressed = levelInfo->Controllers->checkButtonPress(0, WII_BUTTON_A);
173
174 if(levelInfo->Controllers->checkButtonPress(0, WII_BUTTON_B) == true)
175 {
176 // if A and B are pressed at the same time, they want to exit cest menu...
177 levelInfo->Sounds[MENU_YES]->playOnce();
178 doneLooping = true;
179 }
180
181 // Check for the A button to be pressed
182 if(myGameMenu.CurrentButtonState == BUTTON_RELEASED)
183 {
184 myGameMenu.CurrentButtonState = BUTTON_FREE;
185 }
186 else if(myGameMenu.CurrentButtonState == BUTTON_FREE)
187 {
188 myGameMenu.ClickedButtonID = NO_BUTTON_CLICKED;
189 if(menuButtonPressed == true)
190 {
191 myGameMenu.CurrentButtonState = BUTTON_PRESSED;
192 GameMenuMethodsUpdateClickedButtonID(0, &myGameMenu, levelInfo);
193 }
194 }
195 // Once A is pressed, check for it to be released, then once it's released check to see if we were inside
196 // a set of button coordinates...
197 else if((myGameMenu.CurrentButtonState == BUTTON_PRESSED) && (menuButtonPressed == false))
198 {
199 GameMenuMethodsUpdateClickedButtonID(0, &myGameMenu, levelInfo);
200 switch(myGameMenu.ClickedButtonID)
201 {
202 case FINISHED_BUTTON:
203 levelInfo->Sounds[MENU_YES]->playOnce();
204 doneLooping = true;
205 break;
206 case PROJECTILE_BUTTON:
207 if(levelInfo->CurrentUpgradePoints >= gProjectileCost)
208 {
209 levelInfo->Sounds[MENU_YES]->playOnce();
210 levelInfo->CurrentUpgradePoints -= gProjectileCost;
211 levelInfo->CurrentPlayerBulletDamage += 1.0f;
212 gProjectileCost += PROJECTILE_COST_MODIFIER;
213 setupUpgradeWeaponButton(myGameMenu, levelInfo);
214 }
215 else
216 {
217 levelInfo->Sounds[MENU_NO]->playOnce();
218 }
219 break;
220 case ATTACK_BUTTON:
221 if(levelInfo->CurrentUpgradePoints >= gAttackCost)
222 {
223 levelInfo->Sounds[MENU_YES]->playOnce();
224 levelInfo->CurrentUpgradePoints -= gAttackCost;
225 levelInfo->CurrentPlayerRepelDamage += 1.0f;
226 gAttackCost += ATTACK_COST_MODIFIER;
227 setupUpgradeAttackButton(myGameMenu, levelInfo);
228 }
229 else
230 {
231 levelInfo->Sounds[MENU_NO]->playOnce();
232 }
233 break;
234 default:
235 break;
236 }
237
238 myGameMenu.CurrentButtonState = BUTTON_RELEASED;
239 }
240
241 GameMenuMethodsRenderMenu(0, &myGameMenu, levelInfo);
242 }
243 //TODO: We need a dialog that informs the user that the controller is not responding...
244 }
