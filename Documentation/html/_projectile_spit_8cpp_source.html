106void ProjectileSpit::handleSpriteCollisions(LevelInfo *levelInfo, ISprite &sprite, IAudio *soundToPlay, float projectileDamage)
107{
108 LevelMethodCollisionSet collisionSet;
109 std::vector<SpriteBullet>::iterator spriteIterator = myWeaponInfo.Projectiles.begin();
110 for(int idx = 0; idx < myWeaponInfo.CurrentMaxProjectiles; idx++)
111 {
112 // Handle all the controls and inputs for our baddie sprites!
113 if(spriteIterator->isDead() == false)
114 {
115 if(spriteIterator->getLocation().z >= (spriteIterator->getWaterLevel() - 1.0f))
116 {
117 collisionSet = LevelMethodsCheckCollisionSet(myWeaponInfo.Projectiles.at(idx), sprite);
118 if((collisionSet.collisionABIntersects == true) || (collisionSet.collisionACIntersects == true) || (collisionSet.collisionBCIntersects == true))
119 {
120 if ((sprite.getCurrentAnimation() != DEATH_FALLBACKSLOW) && (sprite.getCurrentAnimation() != DEATH_FALLBACK)
121 && (sprite.getCurrentAnimation() != DEATH_FALLFORWARD))
122 {
123 LevelMethodsHandleSparkEffects(levelInfo, &myWeaponInfo.Projectiles.at(idx), 1);
124 // If the sprite isn't dying, collide with the sucka...
125 // If it's not in pain, play the hurty sound, and modify their health...
126 // If we hit both A and B handle a front collision (change movement by 180 degrees)
127 if(collisionSet.collisionABIntersects == true)
128 {
129 spriteIterator->handleCollision(FRONT_ATTACK, 7.0f);
130 spriteIterator->setJump(JUMP_STATE_READY);
131 }
132 // If we hit both A and C handle a rear left collision (change movement by 45 degrees)
133 if(collisionSet.collisionACIntersects == true)
134 {
135 spriteIterator->handleCollision(LEFT_SIDE_ATTACK, 7.0f);
136 spriteIterator->setJump(JUMP_STATE_READY);
137 }
138 // If we hit both B and C handle a rear right collision (change movement by 315 degrees)
139 if(collisionSet.collisionBCIntersects == true)
140 {
141 spriteIterator->handleCollision(RIGHT_SIDE_ATTACK, 7.0f);
142 spriteIterator->setJump(JUMP_STATE_READY);
143 }
144
145 if((sprite.getCurrentAnimation() != PAIN_A) && (sprite.getCurrentAnimation() != PAIN_B)
146 && (sprite.getCurrentAnimation() != PAIN_C))
147 {
148 soundToPlay->playOnce();
149 }
150 sprite.modifyHealthBy(-projectileDamage);
151
152 }
153
154
155 // If a tower killed this sprite, then add experience to that tower
156 ProjectileMethodsHandleIfSpriteDead(sprite, myWeaponInfo);
157
158 }
159 }
160 }
161 spriteIterator++;
162 }

233bool ProjectileSpit::handleTriggerState(PROJECTILE_TRIGGER_STATE state, ISprite *triggeredBySprite, Vector &spriteOffset)
234{
235 bool returnValue = false;
236 // For our tower weapons, we'll use a timer delay to denote their level :)
237 myWeaponInfo.FiringDelayCounter += gTimeScale;
238 if (myWeaponInfo.FiringDelayCounter > myWeaponInfo.MaxFiringDelay)
239 {
240 returnValue = ProjectileMethodsIsWeaponReady(state, triggeredBySprite, myWeaponInfo);
241 if(returnValue == true)
242 {
243 // adjust our angle so that we spread our shots evenly out over the specified range...
244 float angleRange = 45.0;
245 float angleIncrement = (angleRange / (myWeaponInfo.WeaponLevelNumber + 1));
246 float angleOffset = -((myWeaponInfo.WeaponLevelNumber * angleIncrement) / 2);
247
248 // For this projectile, we want to shoot one for every level...
249 for (int idx = 0; idx < myWeaponInfo.WeaponLevelNumber; idx++)
250 {
251 if (myWeaponInfo.Projectiles.at(myWeaponInfo.CurrentProjectileIndex).isDead() == true)
252 {
253 triggeredBySprite->setAnimation(CROUCH_ATTACK);
254 activateProjectile(myWeaponInfo, angleOffset, &myWeaponInfo.Projectiles.at(myWeaponInfo.CurrentProjectileIndex),
255 triggeredBySprite, spriteOffset, myWeaponInfo.ShootingSound);
256
257 angleOffset += angleIncrement;
258
259 ProjectileMethodsIncrementProjectile(myWeaponInfo, MAXIMUM_PROJECTILES);
260
261 myWeaponInfo.FiringDelayCounter = 0;
262 }
263 }
264 ProjectileMethodsReduceExperience(myWeaponInfo, IPROJECTILE_EXP_MODIFIER);
265 }
266 }
267 myWeaponInfo.WeaponState = state;
268 return returnValue;
