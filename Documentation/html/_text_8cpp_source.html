104static void updateTextureCoordinates(Quad *quad, int textureIndex)
105{
106 // Texture our tiles based on the data specified ...
107 ASSERT(textureIndex >= MIN_ASCII_RANGE);
108 ASSERT(textureIndex < MAX_ASCII_RANGE);
109
110 // Figure out our X and Y texture coordinates based on the ASCII value specified in textureIndex
111 float baseTextureX = (textureIndex % NUMBER_OF_CHARACTERS_IN_COLUMN) * WIDTH_OF_EACH_CHARACTER;
112 float baseTextureY = (textureIndex / NUMBER_OF_CHARACTERS_IN_ROW) * HEIGHT_OF_EACH_CHARACTER;
113
114 quad->texture_S[0] = baseTextureX;
115 quad->texture_S[1] = baseTextureX + WIDTH_OF_EACH_CHARACTER;
116 quad->texture_S[2] = baseTextureX + WIDTH_OF_EACH_CHARACTER;
117 quad->texture_S[3] = baseTextureX;
118
119 quad->texture_T[0] = baseTextureY;
120 quad->texture_T[1] = baseTextureY;
121 quad->texture_T[2] = baseTextureY + HEIGHT_OF_EACH_CHARACTER;
122 quad->texture_T[3] = baseTextureY + HEIGHT_OF_EACH_CHARACTER;
123}

164 va_start(args, fmt);

166 // Assign our arguments to a string ...

174 //debugPrint("\n\nASCII VALUE: %d\n\n", asciiValue);usleep(1000000);

181 // Add this quad to our map

183 newQuadIndex++;

187 }
