104static void activateProjectile(int weaponLevel, ISprite *projectile, ISprite *triggeredBySprite, Vector &spriteOffset, IAudio *soundToPlay)
105{
106 float modifyAngleBy = ((45.0f - -45.0f)*((float)rand()/RAND_MAX)) + -45.0f;
107
108 // Determine the level of the bullet we're going to activate then activate and start moving our bullet sprite
109 if(weaponLevel == 1)
110 {
111 projectile->setLevel(0);
112 projectile->setHealth(0.7f);
113 }
114 else if(weaponLevel == 2)
115 {
116 projectile->setLevel(0);
117 projectile->setHealth(1.0f);
118 }
119 else if(weaponLevel == 3)
120 {
121 projectile->setLevel(0);
122 projectile->setHealth(1.1f);
123 }
124 else
125 {
126 projectile->setLevel(0);
127 projectile->setHealth(1.5f);
128 }
129
130 ProjectileMethodsActivate( weaponLevel, 1.0f, modifyAngleBy, projectile, triggeredBySprite, spriteOffset, soundToPlay);
131
132
133 projectile->setVelocities(8.0f, 0.0f, 2.0f);
134 projectile->setControlMagnitude(1.0f);
135 projectile->handleJump(-17.0f);
136}

141void ProjectileBomb::handleSpriteCollisions(LevelInfo *levelInfo, ISprite &sprite, IAudio *soundToPlay, float projectileDamage)
142{
143 LevelMethodCollisionSet collisionSet;
144 std::vector<SpriteBullet>::iterator spriteIterator = myWeaponInfo.Projectiles.begin();
145 for(int idx = 0; idx < myWeaponInfo.CurrentMaxProjectiles; idx++)
146 {
147 // Handle all the controls and inputs for our baddie sprites!
148 if(spriteIterator->isDead() == false)
149 {
150 if(spriteIterator->getLocation().z >= (spriteIterator->getWaterLevel()))
151 {
152 collisionSet = LevelMethodsCheckCollisionSet(*spriteIterator, sprite);
153 if((collisionSet.collisionABIntersects == true) || (collisionSet.collisionACIntersects == true) || (collisionSet.collisionBCIntersects == true))
154 {
155 if ((sprite.getCurrentAnimation() != DEATH_FALLBACKSLOW) && (sprite.getCurrentAnimation() != DEATH_FALLBACK)
156 && (sprite.getCurrentAnimation() != DEATH_FALLFORWARD))
157 {
158 LevelMethodsHandleSparkEffects(levelInfo, &myWeaponInfo.Projectiles.at(idx), 6);
159 // If the sprite isn't dying, collide with the sucka...
160 //spriteIterator->handleCollision(FRONT_COLLISION, 0.1f);
161
162 // We'll have our sprite bounce off of the other sprite if it collides...
163 spriteIterator->handleJump(-17.0f);
164
165 // If it's not in pain, play the hurty sound, and modify their health...
166 if((sprite.getCurrentAnimation() != PAIN_A) && (sprite.getCurrentAnimation() != PAIN_B)
167 && (sprite.getCurrentAnimation() != PAIN_C))
168 {
169 // If we hit both A and B handle a front collision (change movement by 180 degrees)
170 if(collisionSet.collisionABIntersects == true)
171 {
172 //sprite.handleCollision(FRONT_ATTACK, (5.0f * myWeaponInfo.WeaponLevelNumber));
173 sprite.handleJump(-12.0f);
174 }
175 // If we hit both A and C handle a rear left collision (change movement by 45 degrees)
176 if(collisionSet.collisionACIntersects == true)
177 {
178 //sprite.handleCollision(LEFT_SIDE_ATTACK, (5.0f * myWeaponInfo.WeaponLevelNumber));
179 sprite.handleJump(-12.0f);
180 }
181 // If we hit both B and C handle a rear right collision (change movement by 315 degrees)
182 if(collisionSet.collisionBCIntersects == true)
183 {
184 //sprite.handleCollision(RIGHT_SIDE_ATTACK, (5.0f * myWeaponInfo.WeaponLevelNumber));
185 sprite.handleJump(-12.0f);
186 }
187
188 // Play a sound and handle damge
189 soundToPlay->playOnce();
190 sprite.modifyHealthBy(-(projectileDamage * (myWeaponInfo.WeaponLevelNumber * 10)));
191 }
192 }
193
194 // If the player killed this sprite, drop a power up
195 // If a tower killed this sprite, then add experience to that tower
196 ProjectileMethodsHandleIfSpriteDead(sprite, myWeaponInfo);
197 }
198 }// end if >= water level ish
199 }
200 spriteIterator++;
201 }

228bool ProjectileBomb::handleTriggerState(PROJECTILE_TRIGGER_STATE state, ISprite *triggeredBySprite, Vector &spriteOffset)
229{
230
231 // If the sprite animation is right for this type of gun, and the player just released their weapon...
232 bool returnValue = ProjectileMethodsIsWeaponReady(state, triggeredBySprite, myWeaponInfo);
233 if(returnValue == true)
234 {
235 if(myWeaponInfo.Projectiles.at(myWeaponInfo.CurrentProjectileIndex).isDead() == true)
236 {
237 triggeredBySprite->setAnimation(CROUCH_ATTACK);
238
239 activateProjectile(myWeaponInfo.WeaponLevelNumber,
240 &myWeaponInfo.Projectiles.at(myWeaponInfo.CurrentProjectileIndex),
241 triggeredBySprite, spriteOffset, myWeaponInfo.ShootingSound);
242
243 ProjectileMethodsIncrementProjectile(myWeaponInfo, MAXIMUM_PROJECTILES);
244 }
245 }
246 myWeaponInfo.WeaponState = state;
247 return returnValue;
