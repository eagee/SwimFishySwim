97// Used to determine if a vector is inside of a bounding box. Returns true if it is, returns false if not...
98static bool vectorIsInsideBoundingBox(Vector testVector, BoundingBox boundingBox)
99{
100 if((testVector.x >= boundingBox.left) && (testVector.x <= boundingBox.right)
101 && (testVector.y >= boundingBox.top) && (testVector.y <= boundingBox.bottom))
102 {
103 return true;
104 }
105 else
106 {
107 return false;
108 }
109}

111// Used to determine if two vectors intersect a bounding box. Returns true if they do, returns false if not...
112static bool vectorIntersectsBoundingBox(Vector testVectorA, Vector testVectorB, BoundingBox boundingBox)
113{
114 // Check to see if an intersection occurs by always making sure that one of the two points
115 // is within one of the hight/width combinations of the bounding box...
116 if( ((testVectorA.x >= boundingBox.left) && (testVectorA.x <= boundingBox.right) && (testVectorB.y >= boundingBox.top) && (testVectorB.y <= boundingBox.bottom))
117 || ((testVectorB.x >= boundingBox.left) && (testVectorB.x <= boundingBox.right) && (testVectorA.y >= boundingBox.top) && (testVectorA.y <= boundingBox.bottom)))
118 {
119 return true;
120 }
121 else
122 {
123 return false;
124 }

131{
132 DIRECTION returnValue = DIRECTION_NONE;
133
134 // If our joystick is pressed hard enough and in the right angle return true, otherwise false
135 if(magnitude >= 0.25f)

137 if((angle >= 315.0f) || (angle < 45.0f))
138 {
139 returnValue = DIRECTION_NORTH;
140 }
141 else if((angle >= 45.0f) && (angle < 135.0f))

143 returnValue = DIRECTION_EAST;
144 }
145 else if((angle >= 135.0f) && (angle <=225.0f))

147 returnValue = DIRECTION_SOUTH;

150 {
151 returnValue = DIRECTION_WEST;

409bool LevelMethodsHandleInput(int controllerID, CONTROLLER_STATUS &status, LevelInfo *levelInfo)
410{
411 bool doneLooping = false;
412
413 levelInfo->Controllers->updateControllerStates();
414
415 status = levelInfo->Controllers->checkControllerStatus(controllerID);
416
417 if(status == CONTROLLER_READY)
418 {
419 // See if the user wishes to end this level ...
420 if(levelInfo->Controllers->checkButtonPress(controllerID, WII_BUTTON_HOME) == true)
421 {
422 doneLooping = true;
423 }
424
425 //TODO: Remove/Modify these so that cam dist does not change
426 if(levelInfo->Controllers->checkButtonPress(controllerID, WII_BUTTON_UP) == true)
427 {
428 levelInfo->InGameMenu->clickButton(levelInfo, CHOMP_TOWER_BUTTON);
429 //levelInfo->Graphics->adjustCameraDistance(-1.0f);
430 }
431 else if(levelInfo->Controllers->checkButtonPress(controllerID, WII_BUTTON_DOWN) == true)
432 {
433 levelInfo->InGameMenu->clickButton(levelInfo, SPITTING_TOWER_BUTTON);
434 //levelInfo->Graphics->adjustCameraDistance(1.0f);
435 }
436 else if(levelInfo->Controllers->checkButtonPress(controllerID, WII_BUTTON_LEFT) == true)
437 {
438 levelInfo->InGameMenu->clickButton(levelInfo, WIND_TOWER_BUTTON);
439 //levelInfo->Graphics->adjustCameraRadius(1.0f);
440 }
441 else if(levelInfo->Controllers->checkButtonPress(controllerID, WII_BUTTON_RIGHT) == true)
442 {
443 levelInfo->InGameMenu->clickButton(levelInfo, THROWING_TOWER_BUTTON);
444 //levelInfo->Graphics->adjustCameraRadius(-1.0f);
445 }
446 else
447 {
448 // Sending -1 will trigger all buttons to be set as released...
449 levelInfo->InGameMenu->clickButton(levelInfo, NO_TOWER_BUTTON_SELECTED);
450 }
451
452 // Trigger a break point if we hit plus...
453 if(levelInfo->Controllers->checkButtonPress(controllerID, WII_BUTTON_PLUS) == true)
454 {
455#ifdef PLATFORM_IS_WII
456 _break();
457#endif
458 }
459
460 // Handle the trigger state of our current weapon...
461 if (levelInfo->CurrentEnergyLevel >= 1.0f)
462 {
463 if (levelInfo->Controllers->checkButtonPress(controllerID, WII_BUTTON_B) == true)
464 {
465 levelInfo->ProjectileWeapon->handleTriggerState(PROJECTILE_TRIGGER_PRESSED, levelInfo->PlayerOneSprite, levelInfo->SpriteOffset);
466
467 }
468 else
469 {
470 if (levelInfo->ProjectileWeapon->handleTriggerState(PROJECTILE_TRIGGER_RELEASED, levelInfo->PlayerOneSprite, levelInfo->SpriteOffset) == true)
471 {
472 LevelMethodsModifyEnergy(levelInfo, -100.0f);
473 }
474 }
475 }
476
477 // If player presses C on the nunchuck, then cycle through the available weapons...
478 if(levelInfo->Controllers->checkButtonPress(controllerID, WII_BUTTON_C) == true)
479 {
480 gChangeButtonState = CHANGE_WEAPON_PRESSED;
481 }
482 else if (gChangeButtonState == CHANGE_WEAPON_PRESSED)
483 {
484 switchProjectileWeapon(levelInfo, 1);
485 gChangeButtonState = CHANGE_WEAPON_NONE;
486 }
487
488 if(levelInfo->Controllers->checkButtonPress(controllerID, WII_BUTTON_Z) == true)
489 {
490 levelInfo->PlayerOneSprite->handleJump(-20.0f);
491 }
492
493 // Check for a swing to trigger a physical attack...
494 if(levelInfo->Controllers->checkButtonPress(controllerID, WII_HORIZONTAL_SWING_RIGHT) == true)
495 {
496 if((levelInfo->PlayerOneSprite->getCurrentAnimation() != ATTACK_A)
497 && (levelInfo->PlayerOneSprite->getCurrentAnimation() != REVERSE_ATTACK))
498
499 {
500 levelInfo->Sounds[SPIN_ATTACK]->playOnce();
501 levelInfo->PlayerOneSprite->setAnimation(ATTACK_A);
502 LevelMethodsHandleSplashEffects(levelInfo, levelInfo->PlayerOneSprite, 32);
503 LevelMethodsHandleRepelEffects(levelInfo, levelInfo->PlayerOneSprite);
504 }
505
506 }
507 // Check for a swing to trigger a physical attack...
508 else if(levelInfo->Controllers->checkButtonPress(controllerID, WII_HORIZONTAL_SWING_LEFT) == true)
509 {
510 if((levelInfo->PlayerOneSprite->getCurrentAnimation() != ATTACK_A)
511 && (levelInfo->PlayerOneSprite->getCurrentAnimation() != REVERSE_ATTACK))
512 {
513 levelInfo->Sounds[SPIN_ATTACK]->playOnce();
514 levelInfo->PlayerOneSprite->setAnimation(REVERSE_ATTACK);
515 LevelMethodsHandleSplashEffects(levelInfo, levelInfo->PlayerOneSprite, 32);
516 LevelMethodsHandleRepelEffects(levelInfo, levelInfo->PlayerOneSprite);
517 }
518
519 }
520
521
522 if((levelInfo->PlayerOneSprite->getJump() != JUMP_STATE_LANDING))
523 {
524 // Obtain the joystick angle, then modify it to adjust for the camera angle
525 float joystickAngle = levelInfo->Controllers->checkJoystickAngle(controllerID);
526 //printf("\x1b[2;0H"); printf("JoyAngle: %f, CameraAngle: %f\n", joystickAngle, levelInfo->Graphics->getCameraAngle());
527 joystickAngle = adjustInputAngleByCameraAngle(joystickAngle, levelInfo->Graphics->getCameraAngle());
528 //printf("NewAngle: %f", joystickAngle);
529 float joystickMagnitude = levelInfo->Controllers->checkJoystickMagnitude(controllerID);
530
531 if((joystickMagnitude >= 0.250f) && (joystickMagnitude != NAN))
532 {
533 // Adjust our maximum velocity by the joystick magnitude, so that the player can go slower if they want
534 DIRECTION currentDirection = LevelMethodsGetAngleDirection(joystickAngle, joystickMagnitude);
535 levelInfo->PlayerOneSprite->setDirection(currentDirection);
536 if(currentDirection != DIRECTION_NONE)
537 {
538 levelInfo->PlayerOneSprite->increaseAcceleration();
539 }
540 levelInfo->PlayerOneSprite->setAngle(joystickAngle);
541 levelInfo->PlayerOneSprite->setControlMagnitude(joystickMagnitude);
542 }
543 }
544 }
545 //TODO: We need a dialog that informs the user that the controller is not responding...
546 return doneLooping;
547}

716void LevelMethodsHandleTilemapCollision(ISprite &sprite, LevelInfo *levelInfo)
717{
718 float pointAZValue;
719 float pointBZValue;
720 float pointCZValue;
721 float mapZLocation;
722
723 // Get our bounding triangle, sprite location, and map location
724 BoundingTriangle spriteCollision = sprite.getBoundingTriangle();
725 Vector spriteLocation = sprite.getLocation();
726
727 // Get an interpolated Z value for each collision point
728 pointAZValue = getInterpolatedZValue(spriteLocation, spriteCollision.collisionA, *levelInfo->TileMap);
729 pointBZValue = getInterpolatedZValue(spriteLocation, spriteCollision.collisionB, *levelInfo->TileMap);
730 pointCZValue = getInterpolatedZValue(spriteLocation, spriteCollision.collisionC, *levelInfo->TileMap);
731 mapZLocation = getInterpolatedZValue(spriteLocation, VEC_INIT(0.0f, 0.0f, 0.0f), *levelInfo->TileMap);
732
733 if(spriteLocation.z >= sprite.getWaterLevel())
734 {
735 float collisionDistance = -0.25f;
736
737 // Now modify those z values to represent their distance from the sprites water level ...
738 float pointADistance = (pointAZValue - sprite.getWaterLevel());
739 float pointBDistance = (pointBZValue - sprite.getWaterLevel());
740 float pointCDistance = (pointCZValue - sprite.getWaterLevel());
741
742 // If we hit both A and B handle a front collision (change movement by 180 degrees)
743 if((pointADistance < collisionDistance) && (pointBDistance < collisionDistance) && (pointCDistance < collisionDistance))
744 {
745 //LevelMethodsHandleSparkEffects(levelInfo, &sprite, 1);
746 sprite.handleCollision(REAR_COLLISION, 0.0f);
747 }
748 // If we hit both A and B handle a front collision (change movement by 180 degrees)
749 if((pointADistance < collisionDistance) && (pointBDistance < collisionDistance))
750 {
751 //LevelMethodsHandleSparkEffects(levelInfo, &sprite, 1);
752 sprite.handleCollision(FRONT_COLLISION, 0.0f);
753 //sprite.setJump(JUMP_STATE_READY);
754 }
755 // If we hit both A and C handle a rear left collision (change movement by 45 degrees)
756 if((pointADistance < collisionDistance) && (pointCDistance < collisionDistance))
757 {
758 //LevelMethodsHandleSparkEffects(levelInfo, &sprite, 1);
759 sprite.handleCollision(LEFT_SIDE_COLLISION, 0.0f);
760 //sprite.setJump(JUMP_STATE_READY);
761 }
762 // If we hit both B and C handle a rear right collision (change movement by 315 degrees)
763 if((pointBDistance < collisionDistance) && (pointCDistance < collisionDistance))
764 {
765 //LevelMethodsHandleSparkEffects(levelInfo, &sprite, 1);
766 sprite.handleCollision(RIGHT_SIDE_COLLISION, 0.0f);
767 //sprite.setJump(JUMP_STATE_READY);
768 }
769 // If we hit just A handle a front left collision (change movement by 135 degrees)
770 else if(pointADistance < collisionDistance)
771 {
772 //LevelMethodsHandleSparkEffects(levelInfo, &sprite, 1);
773 sprite.handleCollision(FRONT_LEFT_COLLISION, 0.0f);
774 //sprite.setJump(JUMP_STATE_READY);
775 }
776 // If we hit just B handle a front right collision (change movement by 225 degrees)
777 else if(pointBDistance < collisionDistance)
778 {
779 //LevelMethodsHandleSparkEffects(levelInfo, &sprite, 1);
780 sprite.handleCollision(FRONT_RIGHT_COLLISION, 0.0f);
781 //sprite.setJump(JUMP_STATE_READY);
782 }
783 // If we hit just C handle a rear collision (increase forward vector but don't change movement
784 else if(pointCDistance < collisionDistance)
785 {
786 //LevelMethodsHandleSparkEffects(levelInfo, &sprite, 1);
787 sprite.handleCollision(REAR_COLLISION, 0.0f);
788 }
789 }
790 else if((spriteLocation.z < sprite.getWaterLevel()) && (spriteLocation.z >= mapZLocation - 1.0f) )
791 {
792 float furthestAway = mapZLocation;
793 if(pointAZValue > furthestAway) furthestAway = pointAZValue;
794 if(pointBZValue > furthestAway) furthestAway = pointBZValue;
795 if(pointCZValue > furthestAway) furthestAway = pointCZValue;
796
797 if(furthestAway == pointAZValue)
798 {
799 sprite.handleCollision(FRONT_SLIDE, 0.0f);
800 if(mapZLocation >= sprite.getWaterLevel()) sprite.handleJump(-0.1f);
801 }
802 else if(furthestAway == pointBZValue)
803 {
804 sprite.handleCollision(FRONT_SLIDE, 0.0f);
805 if(mapZLocation >= sprite.getWaterLevel()) sprite.handleJump(-0.1f);
806 }
807 else if(furthestAway == pointCZValue)
808 {
809 sprite.handleCollision(REAR_SLIDE, 0.0f);
810 if(mapZLocation >= sprite.getWaterLevel()) sprite.handleJump(-0.1f);
811 }
812 else
813 {
814 levelInfo->PlayerOneSprite->handleJump(-14.0f);
815 }
816 }
817 spriteLocation.z = mapZLocation;
818 sprite.setLocation(spriteLocation);
819
820}

1202LevelMethodCollisionSet LevelMethodsCheckCollisionSet(ISprite &mainSprite, ISprite &otherSprite)
1203{
1204 LevelMethodCollisionSet returnValue;
1205 BoundingBox mainBoundingBox = mainSprite.getBoundingBox();
1206 Vector otherLocation = otherSprite.getLocation();
1207 BoundingTriangle otherBoundingTriangle = otherSprite.getBoundingTriangle();
1208 Vector mainLocation = mainSprite.getLocation();
1209
1210 mainBoundingBox.left += mainLocation.x;
1211 mainBoundingBox.right += mainLocation.x;
1212 mainBoundingBox.top += mainLocation.y;
1213 mainBoundingBox.bottom += mainLocation.y;
1214
1215 // Check point A and point B on the Triangle. If either is inside
1216 // of the bounding box then handle it with a FRONT ATTACK
1217
1218 // Check point A
1219 Vector collisionCheckA;
1220 collisionCheckA.x = (otherLocation.x + otherBoundingTriangle.collisionA.x);
1221 collisionCheckA.y = (otherLocation.y + otherBoundingTriangle.collisionA.y);
1222 returnValue.collisionA = vectorIsInsideBoundingBox(collisionCheckA, mainBoundingBox);
1223
1224 // Check point B
1225 Vector collisionCheckB;
1226 collisionCheckB.x = (otherLocation.x + otherBoundingTriangle.collisionB.x);
1227 collisionCheckB.y = (otherLocation.y + otherBoundingTriangle.collisionB.y);
1228 returnValue.collisionB = vectorIsInsideBoundingBox(collisionCheckB, mainBoundingBox);
1229
1230 // Check point C
1231 Vector collisionCheckC;
1232 collisionCheckC.x = (otherLocation.x + otherBoundingTriangle.collisionC.x);
1233 collisionCheckC.y = (otherLocation.y + otherBoundingTriangle.collisionC.y);
1234 returnValue.collisionC = vectorIsInsideBoundingBox(collisionCheckC, mainBoundingBox);
1235
1236 // Check to see if a line between A and B intersects the bounding box ...
1237 returnValue.collisionABIntersects = vectorIntersectsBoundingBox(collisionCheckA, collisionCheckB, mainBoundingBox);
1238 returnValue.collisionACIntersects = vectorIntersectsBoundingBox(collisionCheckA, collisionCheckC, mainBoundingBox);
1239 returnValue.collisionBCIntersects = vectorIntersectsBoundingBox(collisionCheckB, collisionCheckC, mainBoundingBox);
