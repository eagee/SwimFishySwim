115float fastLinearInterpolation(float valueOne, float valueTwo, float interpolationValue)
116{
117 float returnValue = valueTwo;
118
119 // To perform linear interpolation we find the lowest and higest value between one and two
120 // subtract the lowest from the highest, multiply that value by the interpolation value, and
121 // then add the product to the low value to get our interpolated value ;)
122 // (e.g. low = 5 high = 10 interpolation = 0.5 returnValue = (((10 - 5) * 0.5) + 5) = 7.5)
123
124 if(valueOne > valueTwo)
125 {
126 returnValue = (((valueOne - valueTwo) * interpolationValue) + valueTwo);
127 }
128 else if (valueTwo > valueOne)
129 {
130 returnValue = (((valueOne - valueTwo) * interpolationValue) + valueOne);
131 }
132 return returnValue;
133}
