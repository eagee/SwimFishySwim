142void spriteMethodsHandleCollision(SpriteInfo *spriteInfo, COLLISION_TYPE type, float collisionForce)
143{
144 float angleModification = 0.0f;
145 float forceOfCollision = (spriteInfo->VectorLength * 0.7f);
146
147 switch(type)
148 {
149 case FRONT_COLLISION:
150 angleModification = 180.0f;
151 break;
152 case FRONT_ATTACK:
153 angleModification = 180.0f;
154 forceOfCollision = collisionForce;
155 break;
156 case FRONT_LEFT_COLLISION:
157 angleModification = 225.0f;
158 break;
159 case FRONT_RIGHT_COLLISION:
160 angleModification = 135.0f;
161 break;
162 case LEFT_SIDE_COLLISION:
163 angleModification = 270.0f;
164 break;
165 case RIGHT_SIDE_COLLISION:
166 angleModification = 90.0f;
167 break;
168 case LEFT_SIDE_ATTACK:
169 angleModification = 270.0f;
170 forceOfCollision = collisionForce;
171 break;
172 case RIGHT_SIDE_ATTACK:
173 angleModification = 90.0f;
174 forceOfCollision = collisionForce;
175 break;
176 case REAR_COLLISION:
177 if(spriteInfo->Type == BULLET_SPRITE)
178 {
179 return;
180 }
181 else
182 {
183 forceOfCollision = (spriteInfo->VectorLength * 0.7f);
184 }
185 break;
186 default:
187 ASSERT(false);
188 break;
189 }
190
191 // Whatever our current angle is, we modify it by the type of collision that occurred...
192 float modifiedAngle = modifyAngleByDegrees(spriteInfo->AngleDegrees, angleModification);
193 float angleInRads = getRadiansForAngle(modifiedAngle);
194 spriteInfo->Velocity.x = (forceOfCollision * cos(angleInRads));
195 spriteInfo->Velocity.y = (forceOfCollision * sin(angleInRads));
196 // If this is a bullet sprite, then we'll modify the control angle, too, so that it goes the other way
197 if(spriteInfo->Type == BULLET_SPRITE)
198 {
199 spriteInfo->AngleDegrees = modifiedAngle;
200 spriteInfo->ControlAngle = modifiedAngle - 90.0f;
201 spriteInfo->TargetDegrees = modifiedAngle;
202 if(spriteInfo->ControlAngle < 0.0f) spriteInfo->ControlAngle += 360.0f;
203 spriteInfo->VectorLength *= 2.0f;
204 if(spriteInfo->VectorLength > spriteInfo->MaxVelocity) spriteInfo->VectorLength = spriteInfo->MaxVelocity;
205 }
206 spriteMethodsUpdateLocationByVelocity(spriteInfo);
207}

212void spriteMethodsHandleSlideCollision(SpriteInfo *spriteInfo, COLLISION_TYPE type, float collisionForce)
213{
214 float angleModification = 0.0f;
215
216 //ASSERT(spriteInfo->Type != BULLET_SPRITE);
217
218 // Determine the angle for this type of slide collision..
219 switch (type)
220 {
221 case FRONT_LEFT_SLIDE:
222 angleModification = 90.0f;
223 break;
224 case FRONT_RIGHT_SLIDE:
225 angleModification = 270.0f;
226 break;
227 case REAR_SLIDE:
228 angleModification = 180.0f;
229 break;
230 case FRONT_SLIDE:
231 angleModification = 0.0f;
232 break;
233 default:
234 ASSERT(false);
235 break;
236 }
237
238 // Increase our vector length...
239 if(spriteInfo->VectorLength < (spriteInfo->MaxVelocity)) spriteInfo->VectorLength += spriteInfo->AccelerationVelocity;
240 else spriteInfo->VectorLength = spriteInfo->MaxVelocity;
241
242 // Whatever our current angle is, we modify it by the type of collision that occurred and apply the increased vector to the new direction
243 float modifiedAngle = modifyAngleByDegrees(spriteInfo->AngleDegrees, angleModification);
244 float angleInRads = getRadiansForAngle(modifiedAngle);
245
246 //spriteInfo->Velocity.x = (spriteInfo->VectorLength * cos(angleInRads));
247
248 // Apply these values to our velocity
249 spriteInfo->Velocity.x = (spriteInfo->VectorLength * cos(angleInRads));
250 spriteInfo->Velocity.y = (spriteInfo->VectorLength * sin(angleInRads));
251
252 // And update our position accordingly...
253 spriteMethodsUpdateLocationByVelocity(spriteInfo);
254}

327BoundingTriangle spriteMethodsGetBoundingTriangle(SpriteInfo *spriteInfo)
328{
329 BoundingTriangle returnValue;
330
331 float collisionAngleA = modifyAngleByDegrees(spriteInfo->AngleDegrees, 200);
332 float collisionAngleB = modifyAngleByDegrees(spriteInfo->AngleDegrees, 160);
333 float collisionAngleC = modifyAngleByDegrees(spriteInfo->AngleDegrees, 0);
334
335 // CollisionAngles A to C define a trianged based on our current angle modified by a certain number of degrees.
336 // We will then use the value in spriteInfo->BoundingTriangleVector to convert these angels into cartesian coordinates
337 // that we can then use for collision detection.
338 Vector zeroVector = {0.0f, 0.0f, 0.0f};
339 returnValue.collisionA = modifyVectorByAngle(zeroVector, spriteInfo->BoundingTriangleSize * 1.5f, collisionAngleA);
340 returnValue.collisionB = modifyVectorByAngle(zeroVector, spriteInfo->BoundingTriangleSize * 1.5f, collisionAngleB);
341 returnValue.collisionC = modifyVectorByAngle(zeroVector, (spriteInfo->BoundingTriangleSize * 1.5f), collisionAngleC);
342
343 return returnValue;
344}

350BoundingCircle spriteMethodsGetBoundingCircle(SpriteInfo *spriteInfo, int numberOfPoints)
351{
352 BoundingCircle returnValue;
353
354 float boundingAngle = spriteInfo->AngleDegrees;
355 float angleIncrement = (360.0f / numberOfPoints);
356
357 for(int idx = 0; idx < numberOfPoints; idx++)
358 {
359 BoundingVertex newBoundingVert;
360
361 newBoundingVert.angle = modifyAngleByDegrees(spriteInfo->AngleDegrees, boundingAngle);
362
363 // Front and back of circle should be larger than sides (since all our sprites are generally fish) (yes I know it's not really a circle now)
364 Vector zeroVector = {0.0f, 0.0f, 0.0f};
365 newBoundingVert.vertex = modifyVectorByAngle(zeroVector, spriteInfo->BoundingTriangleSize, newBoundingVert.angle);
366
367 returnValue.verticies.push_back(newBoundingVert);
368
369 // Increment angle by increment degrees (8 points would be 45.0f between each) and adjust for rollover...
370 boundingAngle += angleIncrement;
371 if(boundingAngle > 360.0f)
372 {
373 boundingAngle -= 360;
374 }
375 }
376
377
378 return returnValue;
379}

485void spriteMethodsProcessAnimation(SpriteInfo *spriteInfo)
486{
487 // Change our animation if needed
488 if(spriteInfo->LastAnimation != spriteInfo->CurrentAnimation)
489 {
490 spriteInfo->LastAnimation = spriteInfo->CurrentAnimation;
491 spriteInfo->sprite3dModel->setAnimationType(spriteInfo->CurrentAnimation, spriteInfo->animationState);
492 }
493
494 if(spriteInfo->animationState.currFrame == spriteInfo->animationState.endFrame)
495 {
496 if(((spriteInfo->CurrentAnimation == JUMP) || (spriteInfo->CurrentAnimation == CROUCH_STAND)
497 || (spriteInfo->CurrentAnimation == FLIP) || (spriteInfo->CurrentAnimation == CROUCH_ATTACK)
498 || (spriteInfo->CurrentAnimation == ATTACK_A) || (spriteInfo->CurrentAnimation == REVERSE_ATTACK)
499 || (spriteInfo->CurrentAnimation == PAIN_A) || (spriteInfo->CurrentAnimation == PAIN_B)
500 || (spriteInfo->CurrentAnimation == PAIN_C) || (spriteInfo->CurrentAnimation == JUMP) || (spriteInfo->CurrentAnimation == WAVE)
501 || (spriteInfo->CurrentAnimation == SALUTE)))
502 {
503 if(spriteInfo->Health <= 0.0f)
504 {
505 int deathAnim = (rand() % 3);
506 if(deathAnim == 0) spriteInfo->CurrentAnimation = DEATH_FALLBACK;
507 else if(deathAnim == 1) spriteInfo->CurrentAnimation = DEATH_FALLFORWARD;
508 else if(deathAnim == 2) spriteInfo->CurrentAnimation = DEATH_FALLBACKSLOW;
509 }
510 else
511 {
512 spriteInfo->CurrentAnimation = STAND;
513 }
514 }
515 }
516 else if(spriteInfo->animationState.nextFrame == spriteInfo->animationState.endFrame)
517 {
518 if((spriteInfo->CurrentAnimation == DEATH_FALLBACKSLOW)
519 || (spriteInfo->CurrentAnimation == DEATH_FALLFORWARD)
520 || (spriteInfo->CurrentAnimation == DEATH_FALLBACK)
521 || (spriteInfo->CurrentAnimation == CROUCH_DEATH))
522 {
523 spriteInfo->Dead = true;
524 }
525 }
526
527 // Handle animation here ..
528 spriteInfo->sprite3dModel->handleAnimation(gTimeScale, spriteInfo->animationState);
529}

576void spriteMethodsRenderSprite(SpriteInfo *spriteInfo, IGraphics *graphics, Vector renderOffset, bool interpolate)
577{
578 if(spriteInfo->Dead == false)
579 {
580 // If we're not dying then handle turns
581 if((spriteInfo->CurrentAnimation != DEATH_FALLBACKSLOW)
582 && (spriteInfo->CurrentAnimation != DEATH_FALLFORWARD)
583 && (spriteInfo->CurrentAnimation != DEATH_FALLBACK))
584 {
585 // Adjust our rotation so that currentAngle moves appropriately when targetAngle moves...
586 adjustAngleRotation(spriteInfo->AngleDegrees, spriteInfo->TargetDegrees, (ROTATION_SPEED * spriteInfo->ControlMagnitude));
587
588 // Assign this to our current rotation value...
589 spriteInfo->CurrentRotation.value = spriteInfo->AngleDegrees;
590 }
591
592 // Reset our target distance value on render so that we can accept a new target if a close one dies ...
593 spriteInfo->CurrentTargetDistance = MAXIMUM_TARGET_DISTANCE;
594
595 // Translate and render our sprites model
596 Vector renderLocation = spriteInfo->Location;
597 renderLocation.x += renderOffset.x;
598 renderLocation.y += renderOffset.y;
599 renderLocation.z += renderOffset.z;
600
601 SEEN_BY_CAMERA cameraSight = graphics->isLocationWithinViewingRange(renderLocation);
602 if(cameraSight == SEEN_BY_CAMERA_NEAR)
603 {
604 // and render our model
605 spriteInfo->sprite3dModel->translateModel(renderLocation);
606 graphics->rotateNextObjectInScene(&spriteInfo->BaseRotation);
607 graphics->rotateNextObjectInScene(&spriteInfo->CurrentRotation);
608 spriteInfo->sprite3dModel->renderModel(graphics, spriteInfo->animationState, interpolate);
609 }
610 }
611}

707void spriteMethodsProcessVelocity(SpriteInfo *spriteInfo)
708{
709 if((spriteInfo->CurrentAnimation != ATTACK_A)
710 && (spriteInfo->CurrentAnimation != REVERSE_ATTACK)
711 && (spriteInfo->CurrentAnimation != PAIN_A)
712 && (spriteInfo->CurrentAnimation != PAIN_B)
713 && (spriteInfo->CurrentAnimation != PAIN_C)
714 && (spriteInfo->CurrentAnimation != CROUCH_ATTACK)
715 && (spriteInfo->CurrentAnimation != DEATH_FALLBACKSLOW)
716 && (spriteInfo->CurrentAnimation != DEATH_FALLBACK)
717 && (spriteInfo->CurrentAnimation != DEATH_FALLFORWARD)
718 && (spriteInfo->CurrentAnimation != FLIP)
719 && (spriteInfo->CurrentAnimation != WAVE)
720 && (spriteInfo->CurrentAnimation != SALUTE)
721 && (spriteInfo->Dead == false))
722 {
723 // Always subtract our drag velocity so that if we're
724 // not pressing a key, the fish is slowing down...
725 if(spriteInfo->VectorLength > 0.0f)
726 {
727 if (spriteInfo->CurrentAnimation != JUMP)
728 {
729 if(spriteInfo->VectorLength <= spriteInfo->AccelerationVelocity)
730 {
731 spriteInfo->CurrentAnimation = STAND;
732 }
733 else
734 {
735 spriteInfo->CurrentAnimation = RUN;
736 }
737 }
738 spriteInfo->VectorLength -= spriteInfo->DragVelocity;
