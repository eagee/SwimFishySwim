105void ProjectileMine::handleSpriteCollisions(LevelInfo *levelInfo, ISprite &sprite, IAudio *soundToPlay, float projectileDamage)
106{
107 LevelMethodCollisionSet collisionSet;
108 std::vector<SpriteBullet>::iterator spriteIterator = myWeaponInfo.Projectiles.begin();
109 for(int idx = 0; idx < myWeaponInfo.CurrentMaxProjectiles; idx++)
110 {
111 // Handle all the controls and inputs for our baddie sprites!
112 if(spriteIterator->isDead() == false)
113 {
114 if(spriteIterator->getLocation().z >= (spriteIterator->getWaterLevel()))
115 {
116 collisionSet = LevelMethodsCheckCollisionSet(myWeaponInfo.Projectiles.at(idx), sprite);
117 if((collisionSet.collisionABIntersects == true) || (collisionSet.collisionACIntersects == true) || (collisionSet.collisionBCIntersects == true))
118 {
119 if ((sprite.getCurrentAnimation() != DEATH_FALLBACKSLOW) && (sprite.getCurrentAnimation() != DEATH_FALLBACK)
120 && (sprite.getCurrentAnimation() != DEATH_FALLFORWARD))
121 {
122 LevelMethodsHandleSparkEffects(levelInfo, &myWeaponInfo.Projectiles.at(idx), 30);
123 // If the sprite isn't dying, collide with the sucka...
124 // If it's not in pain, play the hurty sound, and modify their health...
125 // If we hit both A and B handle a front collision (change movement by 180 degrees)
126 if(collisionSet.collisionABIntersects == true)
127 {
128 sprite.handleCollision(FRONT_ATTACK, (5.0f * myWeaponInfo.WeaponLevelNumber));
129 sprite.handleJump(-12.0f);
130 }
131 // If we hit both A and C handle a rear left collision (change movement by 45 degrees)
132 if(collisionSet.collisionACIntersects == true)
133 {
134 sprite.handleCollision(LEFT_SIDE_ATTACK, (5.0f * myWeaponInfo.WeaponLevelNumber));
135 sprite.handleJump(-12.0f);
136 }
137 // If we hit both B and C handle a rear right collision (change movement by 315 degrees)
138 if(collisionSet.collisionBCIntersects == true)
139 {
140 sprite.handleCollision(RIGHT_SIDE_ATTACK, (5.0f * myWeaponInfo.WeaponLevelNumber));
141 sprite.handleJump(-12.0f);
142 }
143
144 if((sprite.getCurrentAnimation() != PAIN_A) && (sprite.getCurrentAnimation() != PAIN_B)
145 && (sprite.getCurrentAnimation() != PAIN_C))
146 {
147 soundToPlay->playOnce();
148 }
149
150 sprite.modifyHealthBy(-(projectileDamage * myWeaponInfo.WeaponLevelNumber));
151
152 // Setup a death animation on the mine that will kill the sprite as well...
153 if(spriteIterator->getLevel() < 4)
154 {
155 spriteIterator->setLevel(4);
156 }
157
158 }
159
160 ProjectileMethodsHandleIfSpriteDead(sprite, myWeaponInfo);
161
162 }
163 }
164 }
165 spriteIterator++;
166 }
167}

191static void activateProjectile(int weaponLevel, float angleOffset, ISprite *projectile, ISprite *triggeredBySprite, Vector &spriteOffset, IAudio *soundToPlay)
192{
193
194 // Determine the level of the bullet we're going to activate then activate and Minet moving our bullet sprite
195 if(weaponLevel == 1)
196 {
197 projectile->setLevel(1);
198 projectile->setHealth(2.7f);
199 }
200 else if(weaponLevel == 2)
201 {
202 projectile->setLevel(1);
203 projectile->setHealth(2.6f);
204 }
205 else if(weaponLevel == 3)
206 {
207 projectile->setLevel(2);
208 projectile->setHealth(2.5f);
209 }
210 else
211 {
212 projectile->setLevel(3);
213 projectile->setHealth(2.4f);
214 }
215
216 ProjectileMethodsActivate( weaponLevel, 1.5, angleOffset, projectile, triggeredBySprite, spriteOffset, soundToPlay);
217
218 projectile->setVelocities(1.0f, 0.0f, 0.001f);
219 projectile->handleJump(-10.0f);
220 projectile->setControlMagnitude(2.0f);
221 projectile->increaseAcceleration();
222
223}

231bool ProjectileMine::handleTriggerState(PROJECTILE_TRIGGER_STATE state, ISprite *triggeredBySprite, Vector &spriteOffset)
232{
233
234 bool returnValue = ProjectileMethodsIsWeaponReady(state, triggeredBySprite, myWeaponInfo);
235 if(returnValue == true)
236 {
237 // For this projectile, we want to shoot one for every level...
238 for (int idx = 0; idx < myWeaponInfo.WeaponLevelNumber; idx++)
239 {
240 if (myWeaponInfo.Projectiles.at(myWeaponInfo.CurrentProjectileIndex).isDead() == true)
241 {
242 triggeredBySprite->setAnimation(CROUCH_ATTACK);
243 activateProjectile(myWeaponInfo.WeaponLevelNumber, 180, &myWeaponInfo.Projectiles.at(myWeaponInfo.CurrentProjectileIndex), triggeredBySprite,
244 spriteOffset, myWeaponInfo.ShootingSound);
245
246 ProjectileMethodsIncrementProjectile(myWeaponInfo, MAXIMUM_PROJECTILES);
247
248 }
249 }
250 }
251 myWeaponInfo.WeaponState = state;
252 return returnValue;
253}
