149 void initialize(BoundingTriangle boundingTriangle)
150 {
151 // top = the lowest number (indicates up)
152 // bottom = the highest number (indicates down)
153 // left = the lowest number (indicates left)
154 // right = the highest number (indicates right)
155 top = 0.0f;
156 bottom = 0.0f;
157 left = 0.0f;
158 right = 0.0f;
159
160 // Find the top y coordinate
161 top = boundingTriangle.collisionA.y;
162 if(boundingTriangle.collisionB.y < top) top = boundingTriangle.collisionB.y;
163 if(boundingTriangle.collisionC.y < top) top = boundingTriangle.collisionC.y;
164
165 // Find the bottom y coordinate
166 bottom = boundingTriangle.collisionA.y;
167 if(boundingTriangle.collisionB.y > bottom) bottom = boundingTriangle.collisionB.y;
168 if(boundingTriangle.collisionC.y > bottom) bottom = boundingTriangle.collisionC.y;
169
170 // Find the left x coordinate
171 left = boundingTriangle.collisionA.x;
172 if(boundingTriangle.collisionB.x < left) left = boundingTriangle.collisionB.x;
173 if(boundingTriangle.collisionC.x < left) left = boundingTriangle.collisionC.x;
174
175 // Find the right x coordinate
176 right = boundingTriangle.collisionA.x;
177 if(boundingTriangle.collisionB.x > right) right = boundingTriangle.collisionB.x;
178 if(boundingTriangle.collisionC.x > right) right = boundingTriangle.collisionC.x;
179
180 // Now we have enough data for our bounding box!
181 // The code can compare x and y coordinates against top, bottom, left, and right
182 }
183} BoundingBox;
