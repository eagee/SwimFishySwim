91void SpriteBullet::handleCollision(COLLISION_TYPE type, float collisionForce)
92{
93 // If we're in the water, we want to use the standard collision handler
94 if (type < FRONT_LEFT_SLIDE)
95 {
96 spriteMethodsHandleCollision(&mySpriteInfo, type, collisionForce);
97 }
98 // If we're stuck up on land then we want to use the sliding collision handler...
99 else
100 {
101 spriteMethodsHandleSlideCollision(&mySpriteInfo, type, collisionForce);
102 }
103
104 handleJump(-12.0f);
105
106 if(mySpriteInfo.SpriteLevel <= 0)
107 {
108 mySpriteInfo.Health -= (mySpriteInfo.Health/2);
109 }
110 else if(mySpriteInfo.SpriteLevel == 1)
111 {
112 mySpriteInfo.Health -= (mySpriteInfo.Health/2);
113 }
114 else if(mySpriteInfo.SpriteLevel == 2)
115 {
116 mySpriteInfo.Health -= (mySpriteInfo.Health/2);
117 }
118 else if(mySpriteInfo.SpriteLevel == 3)
119 {
120 mySpriteInfo.Health -= (mySpriteInfo.Health/8);
121 }
122}

147Vector SpriteBullet::updateLocationByVelocity()
148{
149 // Reduce our health as we move ...
150 mySpriteInfo.Health -= 0.01f;
151
152 if(mySpriteInfo.Health <= 0.5f)
153 {
154
155 //myBulletSize = 0.0f;
156 if(mySpriteInfo.Health <= -0.1f)
157 {
158 mySpriteInfo.Dead = true;
159 }
160 mySpriteInfo.Velocity.x /= 1.1f;
161 mySpriteInfo.Velocity.y /= 1.1f;
162 mySpriteInfo.VectorLength /= 1.1f;
163 }
164
165 return spriteMethodsUpdateLocationByVelocity(&mySpriteInfo);
166}

430void SpriteBullet::setLevel(int level)
431{
432 ASSERT(mySpriteInfo.sprite3dModel != NULL);
433
434 mySpriteInfo.SpriteLevel = level;
435
436
437 // Set up the method by which we will display our bullet...
438 if(level == 0)
439 {
440 mySpriteInfo.CurrentAnimation = PAIN_A;
441 mySpriteInfo.sprite3dModel->setAnimationType(PAIN_A, mySpriteInfo.animationState);
442 }
443 else if (level == 1)
444 {
445 mySpriteInfo.CurrentAnimation = PAIN_B;
446 mySpriteInfo.sprite3dModel->setAnimationType(PAIN_B, mySpriteInfo.animationState);
447 }
448 else if (level == 2)
449 {
450 mySpriteInfo.CurrentAnimation = PAIN_C;
451 mySpriteInfo.sprite3dModel->setAnimationType(PAIN_C, mySpriteInfo.animationState);
452 }
453 else if (level == 3)
454 {
455 mySpriteInfo.CurrentAnimation = CROUCH_PAIN;
456 mySpriteInfo.sprite3dModel->setAnimationType(CROUCH_PAIN, mySpriteInfo.animationState);
457 }
458 else if (level == 4)
459 {
460 mySpriteInfo.CurrentAnimation = DEATH_FALLBACK;
461 mySpriteInfo.sprite3dModel->setAnimationType(DEATH_FALLBACK, mySpriteInfo.animationState);
462 }
463}
