567void ModelMd2::buildInterpolatedFrame(TriangleVertexArray &currentFrame, TriangleVertexArray &nextFrame, TriangleVertexArray &interpolatedFrame)
568{
569 // Loop through the number of traingles used in this model ...
570 for(u32 triangleIndex = 0; triangleIndex < (myMd2HeaderData.numTriangles * 3); triangleIndex+=3)
571 {
572 // To interpolate each triangle we:
573 // 1. subtract the verticies from the next frame from the current frame and store the difference in the verticies for the interpolate frame
574 // 2. Multiply the value stored in interpolated frame by the interpolation value
575 // 3. Add current frame to the value in interpolated frame, and voila! We've got an interpolated triangle
576
577 fastVectorSubtract(&nextFrame.Verticies[triangleIndex + 0], &currentFrame.Verticies[triangleIndex + 0], &interpolatedFrame.Verticies[triangleIndex + 0]);
578 fastVectorSubtract(&nextFrame.Verticies[triangleIndex + 1], &currentFrame.Verticies[triangleIndex + 1], &interpolatedFrame.Verticies[triangleIndex + 1]);
579 fastVectorSubtract(&nextFrame.Verticies[triangleIndex + 2], &currentFrame.Verticies[triangleIndex + 2], &interpolatedFrame.Verticies[triangleIndex + 2]);
580
581 fastVectorScale(&interpolatedFrame.Verticies[triangleIndex + 0], &interpolatedFrame.Verticies[triangleIndex + 0], myAnim.interpolateBy);
582 fastVectorScale(&interpolatedFrame.Verticies[triangleIndex + 1], &interpolatedFrame.Verticies[triangleIndex + 1], myAnim.interpolateBy);
583 fastVectorScale(&interpolatedFrame.Verticies[triangleIndex + 2], &interpolatedFrame.Verticies[triangleIndex + 2], myAnim.interpolateBy);
584
585 fastVectorAdd(&interpolatedFrame.Verticies[triangleIndex + 0], &currentFrame.Verticies[triangleIndex + 0], &interpolatedFrame.Verticies[triangleIndex + 0]);
586 fastVectorAdd(&interpolatedFrame.Verticies[triangleIndex + 1], &currentFrame.Verticies[triangleIndex + 1], &interpolatedFrame.Verticies[triangleIndex + 1]);
587 fastVectorAdd(&interpolatedFrame.Verticies[triangleIndex + 2], &currentFrame.Verticies[triangleIndex + 2], &interpolatedFrame.Verticies[triangleIndex + 2]);
588 }
589
590#ifdef PLATFORM_IS_WII
591 DCFlushRange(interpolatedFrame.Verticies, (interpolatedFrame.VertexStride * interpolatedFrame.VertexArraySize * 3));
592#endif
593}

895void ModelMd2::handleAnimation(float time, ModelAnimState &modelAnim)
896{
897 modelAnim.currentTime += time;
898
899 float timeDifference = (modelAnim.currentTime - modelAnim.lastTime);
900 float animFPS = (1.0f / modelAnim.fps);
901 modelAnim.frameIncrement += (timeDifference);
902
903 if ((myAnimationFormat == ANIMATION_PRE_RENDERED_MD2) || (myAnimationFormat == ANIMATION_BADDIE_MD2))
904 {
905 // If enough time has passed to render a single frame...
906 if ((modelAnim.frameIncrement > animFPS) || (modelAnim.lastTime == 0.0f))
907 {
908 modelAnim.frameIncrement -= animFPS;
909
910 modelAnim.currFrame = modelAnim.nextFrame;
911
912 //while (modelAnim.frameIncrement >= animFPS)
913 //{
914 // Reduce the frame increment by one frame, and change our frame number
915 // either backward or forward based on the animation start and end frames
916 modelAnim.frameIncrement -= animFPS;
917
918 // Ok, this is how this works. The max # of frames we can skip for an animation
919 // is the fps for that animation / 60. For instance:
920 // [fps == 60] maxFrameSkip = 1;
921 // [fps == 30] maxFrameSkip = 0; (30 FPS in theory is the lowest we will go due to vsync)
922 // [fps == 480] maxFrameSkip = 8;
923 int maxFrameSkip = (modelAnim.fps / 60);
924 modelAnim.frameSkip = (char)(modelAnim.frameIncrement / animFPS);
925 if(modelAnim.frameSkip > maxFrameSkip) modelAnim.frameSkip = maxFrameSkip;
926
927 if (modelAnim.startFrame <= modelAnim.endFrame)
928 {
929 modelAnim.nextFrame += (1 + modelAnim.frameSkip);
930 if (modelAnim.nextFrame > modelAnim.endFrame)
931 {
932 modelAnim.currFrame = modelAnim.endFrame;
933 modelAnim.nextFrame = modelAnim.startFrame;
934 modelAnim.frameLoopCount++;
935 modelAnim.frameIncrement = 0.0f;
936 }
937 }
938 else
939 {
940 modelAnim.nextFrame -= (1 + modelAnim.frameSkip);
941 if (modelAnim.nextFrame <= modelAnim.endFrame)
942 {
943 modelAnim.currFrame = modelAnim.endFrame;
944 modelAnim.nextFrame = modelAnim.startFrame;
945 modelAnim.frameLoopCount++;
946 modelAnim.frameIncrement = 0.0f;
947 }
948 }
949 //} // end while
950 }
951 modelAnim.lastTime = modelAnim.currentTime;
952 }
953 else //if (myAnimationFormat == ANIMATION_TRADITIONAL_MD2), or STATIC
954 {
955 if ((timeDifference > animFPS) || (modelAnim.lastTime == 0.0f))
956 {
957 modelAnim.currFrame = modelAnim.nextFrame;
958
959 if (modelAnim.startFrame <= modelAnim.endFrame)
960 {
961 modelAnim.nextFrame++;
962 if (modelAnim.nextFrame > modelAnim.endFrame)

964 modelAnim.currFrame = modelAnim.endFrame;

966 }

971 if (modelAnim.nextFrame <= modelAnim.endFrame)
972 {

976 }
977
978 modelAnim.lastTime = modelAnim.currentTime;
979 }
980 }
981
982 // Prevent ouverrunning our total frames
983 if (modelAnim.currFrame > (myMd2HeaderData.numFrames - 1))
984 {
985 modelAnim.currFrame = 0;
986 }
987
988 if (modelAnim.nextFrame > (myMd2HeaderData.numFrames - 1))
989 {
990 modelAnim.nextFrame = 0;
991 }
992
993 // Round off our interpolation value to the nearest 10th...

995 //modelAnim.interpolateBy = float(int(modelAnim.interpolateBy * 10 + 0.5)) / 10;
