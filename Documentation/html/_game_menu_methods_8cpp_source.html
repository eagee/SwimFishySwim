142void GameMenuMethodsSetupButton(Vector location, LevelInfo *levelInfo, GameButton *button, float w, float h, std::string text)
143{
144 button->label.width = DEFAULT_TEXT_WIDTH;
145 button->label.height = DEFAULT_TEXT_HEIGHT;
146 button->label.spacing = DEFAULT_TEXT_SPACING;
147
148 setupBasicQuadDetails(button->quad);
149
150 float adjustedW;
151 float adjustedH;
152 float adjustedOffset = 0.0f;
153
154 adjustedW = levelInfo->TextOutput->getTextSpacing() * text.length();
155 adjustedH = levelInfo->TextOutput->getTextHeight();
156
157 // If the width or height of the button is greater than the specified adjustment, then make sure we use the specified height
158 if(h > adjustedH)
159 {
160 adjustedH = h;
161 }
162 if(w > adjustedW)
163 {
164 adjustedOffset = (w - adjustedW) / 2.0f;
165 adjustedW = w;
166 }
167
168 button->quad.verticies[0] = VEC_INIT(location.x, location.y, location.z);
169 button->quad.verticies[1] = VEC_INIT(location.x + adjustedW, location.y, location.z);
170 button->quad.verticies[2] = VEC_INIT(location.x + adjustedW, location.y + adjustedH, location.z);
171 button->quad.verticies[3] = VEC_INIT(location.x, location.y + adjustedH, location.z);
172
173 location.z -= 0.001f;
174 location.x += adjustedOffset;
175 button->label.location = location;
176 button->label.text = text;
177}

230void GameMethodsRenderButtonsOnly(int controllerID, GameMenuInfo *menuInfo, LevelInfo *levelInfo)
231{
232 std::map<int, Quad> buttonMap;
233
234 // Get our mouse pointer information...
235 Vector pointerLocation = getPointerLocation(controllerID, levelInfo);
236
237 // Add the buttons from our gamemenuinfo object to our quad map
238 std::map<int, GameButton>::iterator buttonIterator = menuInfo->ButtonList.begin();
239 for (u32 idx = 0; idx < menuInfo->ButtonList.size(); idx++)
240 {
241 // If our pointer is within the quad, then create a new quad and make it bigger, otherwise
242 // just assign it. Also, don't make it bigger if it's being clicked (the small will look like a click, then)
243 ASSERT(buttonIterator != menuInfo->ButtonList.end());
244
245 Quad buttonQuad = buttonIterator->second.quad;
246
247 if (buttonIterator->second.enabled == ENABLED_STATE_ENABLED)
248 {
249 // Increase the size of our button if it has focus...
250 if ((pointerLocation.x > buttonQuad.verticies[0].x) && (pointerLocation.x < buttonQuad.verticies[1].x) && (pointerLocation.y
251 > buttonQuad.verticies[0].y) && (pointerLocation.y < buttonQuad.verticies[3].y) && (menuInfo->ClickedButtonID != buttonIterator->second.ID))
252 {
253 buttonQuad.verticies[0].x -= 0.05f;
254 buttonQuad.verticies[1].x += 0.05f;
255 buttonQuad.verticies[2].x += 0.05f;
256 buttonQuad.verticies[3].x -= 0.05f;
257
258 buttonQuad.verticies[0].y -= 0.05f;
259 buttonQuad.verticies[1].y -= 0.05f;
260 buttonQuad.verticies[2].y += 0.05f;
261 buttonQuad.verticies[3].y += 0.05f;
262 }
263 setupEnabledButtonTexture(buttonQuad);
264 }
265 else if (buttonIterator->second.enabled == ENABLED_STATE_ALMOST)
266 {
267 int randEnabled = (rand() % 5);
268 if(randEnabled <= 3)
269 {
270 setupDisabledButtonTexture(buttonQuad);
271 }
272 else
273 {
274 setupEnabledButtonTexture(buttonQuad);
275 }
276 }
277 else if (buttonIterator->second.enabled == ENABLED_STATE_JUST_USED)
278 {
279 int randEnabled = (rand() % 3);
280 if (randEnabled <= 1)
281 {
282 setupEnabledButtonTexture(buttonQuad);
283 }
284 else
285 {
286 setupDisabledButtonTexture(buttonQuad);
287 }
288 }
289 else
290 {
291 setupDisabledButtonTexture(buttonQuad);
292 }
293
294 // Handle rendering the buttons one at a time
295 buttonMap[idx] = buttonQuad;
296 levelInfo->Graphics->addQuadsToScene(VEC_INIT(0.0f,0.0f,0.0f),
297 buttonMap, buttonMap.size(),
298 levelInfo->TowerButtonTextures[buttonIterator->second.ID]);
299 buttonMap.clear();
300
301 buttonIterator++;
302
303 }
304
305 // Print all of the labels for our buttons...
306 std::map<int, GameButton>::iterator buttonListIterator = menuInfo->ButtonList.begin();
307 for (u32 idx = 0; idx < menuInfo->ButtonList.size(); idx++)
308 {
309 levelInfo->TextOutput->setTextSize(buttonListIterator->second.label.width, buttonListIterator->second.label.height,
310 buttonListIterator->second.label.spacing);
311 levelInfo->TextOutput->renderText(buttonListIterator->second.label.location, "%s", buttonListIterator->second.label.text.c_str());
312 buttonListIterator++;
313 }
314
315 // Draw our radar/mouse pointer here ...
316 //levelInfo->TextOutput->renderText(pointerLocation, "^ %f,%f", pointerLocation.x, pointerLocation.y);
317 levelInfo->TextOutput->renderText(pointerLocation, "^");
318 GameMenuMethodsRenderPointer(levelInfo, pointerLocation);
319
320 // Put our text back to the default size
321 levelInfo->TextOutput->setTextSize(DEFAULT_TEXT_WIDTH, DEFAULT_TEXT_HEIGHT, DEFAULT_TEXT_SPACING);
322
323 buttonMap.clear();
324}
