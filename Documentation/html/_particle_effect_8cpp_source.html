98{
99 particle->lifeSpan = SPARK_LIFE_SPAN;

101 // The minus at the end of the mod rand is half of the max result (so that it can go

117static void setupSplashEffect(Particle *particle, Vector location,
118 ISprite *sprite, float &particleDegrees, float &particleIncrement)
119{
120 AnimationType animation = sprite->getCurrentAnimation();
121
122 if ((animation != ATTACK_A) && (animation != REVERSE_ATTACK) && (animation != SALUTE) && (animation != STAND))
123 {
124 particle->lifeSpan = sprite->getSpeed() * 0.5f;//SPLASH_LIFE_SPAN;
125 particle->scale = sprite->getSpeed() * 0.14f;
126 }
127 else
128 {
129 particle->lifeSpan = 0.4f;//SPLASH_LIFE_SPAN;
130 particle->scale = 0.1f;
131 }
132
133 float zSpeed = sprite->getSpeed();
134 if (zSpeed <= 0.5f)
135 {
136 zSpeed = 0.5f;
137 }
138 else if (zSpeed >= 0.5f)
139 {
140 zSpeed = 0.5f;
141 }
142
143 // Particles need to either move left or right of current velocity
144 // because we want our splash to look like a wake when things are moving
145 // in the water :)
146 // If we're creating a salute splash (big one!)
147 if ((animation != ATTACK_A) && (animation != REVERSE_ATTACK) && (animation != SALUTE) && (animation != STAND))
148 {
149 // Assign a random velocity to each spark (a spark can fly off in any direction)
150 // The minus at the end of the mod rand is half of the max result (so that it can go
151 // either direction)
152 particle->velocity = sprite->getVelocity();
153 particle->velocity.x = -(particle->velocity.x * gTimeScale) / 7;
154 particle->velocity.y = -(particle->velocity.y * gTimeScale) / 7;
155
156 if ((rand() % 2) == 0)
157 {
158 // get our new angle 45 degrees away from the direction the sprite is heading
159 float newAngle = sprite->getAngle();
160 newAngle = modifyAngleByDegrees(newAngle, 122.5f);
161
162 Vector newLocation = modifyVectorByAngle(location, 1.5f, newAngle);
163
164 particle->velocity.x += (newLocation.x - location.x) * (gTimeScale * 2);
165 particle->velocity.y += (newLocation.y - location.y) * (gTimeScale * 2);
166 particle->location = newLocation;
167
168 }
169 else
170 {
171 // get our new angle 45 degrees away from the direction the sprite is heading
172 float newAngle = sprite->getAngle();
173 newAngle = modifyAngleByDegrees(newAngle, -122.5f);
174
175 Vector newLocation = modifyVectorByAngle(location, 1.5f, newAngle);
176
177 particle->velocity.x += (newLocation.x - location.x) * (gTimeScale * 2);
178 particle->velocity.y += (newLocation.y - location.y) * (gTimeScale * 2);
179 particle->location = newLocation;
180 }
181 particle->velocity.z = -zSpeed * (gTimeScale * 12);
182 }
183 // If we're creating an ATTACK or SALUTE splash :)
184 else
185 {
186 Vector newLocation;
187 // Assign a random velocity to each spark (a spark can fly off in any direction)
188 // The minus at the end of the mod rand is half of the max result (so that it can go
189 // either direction)
190 particle->velocity = sprite->getVelocity();
191 particle->velocity.x = -(particle->velocity.x * gTimeScale) / 7;
192 particle->velocity.y = -(particle->velocity.y * gTimeScale) / 7;
193
194 // Have our angle increment so that a circle is made out of the
195 // available particles that seems to map with the attack mode...
196 if (animation == SALUTE)
197 {
198 particleDegrees = modifyAngleByDegrees(particleDegrees, particleIncrement);
199 newLocation = modifyVectorByAngle(location, fmod((float)rand(), 2.6f), particleDegrees);
200 particle->location = newLocation;
201 }
202 else if (animation == STAND)
203 {
204 particleDegrees = modifyAngleByDegrees(particleDegrees, particleIncrement);
205 newLocation = modifyVectorByAngle(location, fmod((float)rand(), 4.0f), particleDegrees);
206 particle->location = newLocation;
207 }
208 else if (animation == ATTACK_A)
209 {
210 particleDegrees = modifyAngleByDegrees(particleDegrees, particleIncrement);
211 newLocation = modifyVectorByAngle(location, 2.60f, particleDegrees);
212 particle->location = newLocation;
213 }
214 else
215 {
216 particleDegrees = modifyAngleByDegrees(particleDegrees, -particleIncrement);
217 newLocation = modifyVectorByAngle(location, 2.60f, particleDegrees);
218 particle->location = newLocation;
219 }
220
221 if(animation == SALUTE)
222 {
223 particle->velocity.z = -zSpeed * (0.016f * 65.0f);
224
225 }
226 if(animation == STAND)
227 {
228 particle->velocity.z = -zSpeed * (0.016f * 20.0f);
229 }
230 else
231 {
232 particle->velocity.z = -zSpeed * (0.016f * 30.0f);
233 }
234 particle->velocity.x += (newLocation.x - location.x) * (gTimeScale * 2);
235 particle->velocity.y += (newLocation.y - location.y) * (gTimeScale * 2);
236 }
237
238 particle->velocity.x += (fmod((float)rand(), 32.0f) * 0.001f) - 0.016f;
239 particle->velocity.y += (fmod((float)rand(), 32.0f) * 0.001f) - 0.016f;
240 particle->velocity.z += (fmod((float)rand(), 8.0f) * 0.01f) - 0.04f;
241}

247void ParticleEffect::addParticles(ISprite *sprite, Vector offset, int number)
248{
249 // iterate through our particle list and assign as many particles of the specified
250 // type as possible (depending on whether or not there is a slot with a dead particle open...
251
252 int numberAssigned = 0;
253 float delayValue = 0;
254
255 std::vector<Particle>::iterator particleIterator;
256 Vector location = sprite->getLocation();
257
258 location.x -= offset.x;
259 location.y -= offset.y;
260 location.z = sprite->getWaterLevel();
261
262 myParticleDegrees = sprite->getAngle();
263 myParticleIncrement = (360.0f / (float)number);
264
265 particleIterator = myParticlePool.begin();
266 while ((numberAssigned < number) && (particleIterator != myParticlePool.end()))
267 {
268 if (particleIterator->lifeSpan <= 0)
269 {
270 if (myEffectType == PARTICLE_EFFECT_WATER_RIPPLE)
271 {
272 particleIterator->delay = 0;
273 setupWaterRipple(&*particleIterator, location);
274 }
275 else if (myEffectType == PARTICLE_EFFECT_REPEL_ENEMIES)
276 {
277 particleIterator->delay = 0;
278 setupRepelEffect(&*particleIterator, location, sprite);
279 }
280 else if (myEffectType == PARTICLE_EFFECT_WARP)
281 {
282 particleIterator->delay = 0;
283 setupWarpEffect(&*particleIterator, location, sprite);
284 }
285 else if (myEffectType == PARTICLE_EFFECT_SPARK)
286 {
287 particleIterator->delay = 0;
288 setupSparkEffect(&*particleIterator, location);
289 }
290 else if (myEffectType == PARTICLE_EFFECT_SPLASH)
291 {
292 delayValue += 0.5f;
293 particleIterator->delay = (int)delayValue;
294 setupSplashEffect(&*particleIterator, location,
295 sprite, myParticleDegrees, myParticleIncrement);
296 } // end setup splash...
297 numberAssigned++;
298 }
299 particleIterator++;
300 }
301}

307void ParticleEffect::updateParticles()
308{
309 // Iterate through our particles and update their life span and locations
310 std::vector<Particle>::iterator particleIterator;
311
312 particleIterator = myParticlePool.begin();
313 while(particleIterator != myParticlePool.end())
314 {
315 if(particleIterator->delay > 0)
316 {
317 particleIterator->delay -= 3;
318 }
319 else if(particleIterator->lifeSpan > 0)
320 {
321 // Water ripple doesn't change it's location...
322 if(myEffectType == PARTICLE_EFFECT_WATER_RIPPLE)
323 {
324 particleIterator->lifeSpan -= (1.0f * gTimeScale);
325 particleIterator->scale += (10.0f * gTimeScale);
326 particleIterator->location.z += 0.0001f;
327
328 }
329 else if(myEffectType == PARTICLE_EFFECT_REPEL_ENEMIES)
330 {
331 particleIterator->lifeSpan -= (0.8f * gTimeScale);
332 if(particleIterator->lifeSpan > (REPEL_LIFE_SPAN/2))
333 {
334 particleIterator->scale += (30.0f * gTimeScale);
335 }
336 else
337 {
338 particleIterator->scale -= (40.0f * gTimeScale);
339 }
340 particleIterator->location.z += 0.0001f;
341 particleIterator->location.x += -(particleIterator->velocity.x * gTimeScale);
342 particleIterator->location.y += -(particleIterator->velocity.y * gTimeScale);
343 }
344 else if (myEffectType == PARTICLE_EFFECT_WARP)
345 {
346 particleIterator->lifeSpan -= (1.6f * gTimeScale);
347 if (particleIterator->lifeSpan > (WARP_LIFE_SPAN / 2))
348 {
349 particleIterator->scale += (60.0f * gTimeScale);
350 }
351 else
352 {
353 particleIterator->scale -= (80.0f * gTimeScale);
354 }
355 particleIterator->location.z -= 0.01f;
356 particleIterator->location.x += -(particleIterator->velocity.x * gTimeScale);
357 particleIterator->location.y += -(particleIterator->velocity.y * gTimeScale);
358 }
359 else if (myEffectType == PARTICLE_EFFECT_SPARK)
360 {
361 particleIterator->lifeSpan -= (1.0f * gTimeScale);
362 particleIterator->scale = particleIterator->lifeSpan;
363 particleIterator->location.x += particleIterator->velocity.x;

366 //particleIterator->velocity.z += 0.005f;
367 }
368 else if (myEffectType == PARTICLE_EFFECT_SPLASH)
369 {
370 particleIterator->lifeSpan -= (1.0f * gTimeScale);
371 particleIterator->location.x += particleIterator->velocity.x;
372 particleIterator->location.y += particleIterator->velocity.y;
373 particleIterator->location.z += particleIterator->velocity.z;
374
375 if(particleIterator->location.z <= SPRITE_WATER_LEVEL)
376 {
377 particleIterator->velocity.z += 0.0235f;
378 particleIterator->scale += 1.0f * gTimeScale;

381 {
382 particleIterator->velocity.z -= 0.0470f;
383 particleIterator->scale -= 4.0f * gTimeScale;

394static Quad setupQuad(PARTICLE_EFFECT_TYPE effectType, Particle particle)
395{
396 Quad quad;
397
398 // Set up the verticies for quad so that we're drawing a health bar on top of the sprite
399 // and in the center...
400
401 if((effectType == PARTICLE_EFFECT_WATER_RIPPLE) || (effectType == PARTICLE_EFFECT_REPEL_ENEMIES) || (effectType == PARTICLE_EFFECT_WARP))
402 {
403 quad.verticies[0].x = -(particle.scale / 2);
404 quad.verticies[0].y = -(particle.scale / 2);
405 quad.verticies[0].z = 0.0f;
406
407 quad.verticies[1].x = (particle.scale / 2);
408 quad.verticies[1].y = -(particle.scale / 2);
409 quad.verticies[1].z = 0.0f;
410
411 quad.verticies[2].x = (particle.scale / 2);
412 quad.verticies[2].y = (particle.scale / 2);
413 quad.verticies[2].z = 0.0f;
414
415 quad.verticies[3].x = -(particle.scale / 2);
416 quad.verticies[3].y = (particle.scale / 2);
417 quad.verticies[3].z = 0.0f;
418 }
419 else if((effectType == PARTICLE_EFFECT_SPARK) || (effectType == PARTICLE_EFFECT_SPLASH))
420 {
421 quad.verticies[0].x = -(particle.scale / 2);
422 quad.verticies[0].y = 0.0f;
423 quad.verticies[0].z = -(particle.scale / 2);
424
425 quad.verticies[1].x = (particle.scale / 2);
426 quad.verticies[1].y = 0.0f;
427 quad.verticies[1].z = -(particle.scale / 2);
428
429 quad.verticies[2].x = (particle.scale / 2);
430 quad.verticies[2].y = 0.0f;
431 quad.verticies[2].z = (particle.scale / 2);
432
433 quad.verticies[3].x = -(particle.scale / 2);
434 quad.verticies[3].y = 0.0f;
435 quad.verticies[3].z = (particle.scale / 2);
436 }
437
438 // TopLeft, TopRight, BottomRight, BottomLeft tex coords particle

440 quad.texture_T[0] = 0.0f;
441
442 quad.texture_S[1] = 1.0f;
443 quad.texture_T[1] = 0.0f;
444
445 quad.texture_S[2] = 1.0f;
446 quad.texture_T[2] = 1.0f;
447
448 quad.texture_S[3] = 0.0f;
449 quad.texture_T[3] = 1.0f;
450
451 // That should be everything, the quad should be all set up...
452 return quad;
453}

458void ParticleEffect::renderParticles(Vector offset, IGraphics *graphics)
459{
460
461 ASSERT(myTextureID != UNDEFINED_TEXTURE_HANDLE);
462
463 // Iterate through our particles and render them!
464 std::vector<Particle>::iterator particleIterator;
465
466 particleIterator = myParticlePool.begin();
467 while (particleIterator != myParticlePool.end())
468 {
469 if (particleIterator->lifeSpan > 0)
470 {
471 std::map<int, Quad> quads;
472 Quad renderingQuad = setupQuad(myEffectType, *particleIterator);
473 quads[0] = renderingQuad;
474 Vector renderingLocation;
475 renderingLocation.x = (particleIterator->location.x + offset.x);
476 renderingLocation.y = (particleIterator->location.y + offset.y);
477 renderingLocation.z = (particleIterator->location.z + offset.z);
478
479 // If it's one of the below types, then rotate the particle toward the camera
480 if((myEffectType == PARTICLE_EFFECT_SPARK) || (myEffectType == PARTICLE_EFFECT_SPLASH))
481 {
482
483 Rotate rotation = {{0.0f, 0.0f, 1.0f}, graphics->getCameraAngle() };
484 graphics->rotateNextObjectInScene(&rotation);
485 }
486
487 graphics->addQuadsToScene(renderingLocation, quads, quads.size(), myTextureID);
488 quads.clear();
489 }
490 particleIterator++;
491 }
492}
