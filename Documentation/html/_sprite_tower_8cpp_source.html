500void SpriteTower::lockOnTarget(ISprite &target)
501{
502 bool lockedOntoTarget = spriteMethodsLockOnTarget(target, &mySpriteInfo);
503 if(lockedOntoTarget == true)
504 {
505 // Since we have a baddie in view, trigger our weapon. The weapon will handle firing rate based on tower sprite animation...
506 ASSERT(mySpriteInfo.spriteWeapon != NULL);
507 Vector spriteOffset = myOffsetValue;
508 spriteOffset.z = -3.0f;
509 mySpriteInfo.spriteWeapon->handleTriggerState(PROJECTILE_TRIGGER_PRESSED, this, spriteOffset);
510 mySpriteInfo.spriteWeapon->handleTriggerState(PROJECTILE_TRIGGER_RELEASED, this, spriteOffset);
511 }
512}
