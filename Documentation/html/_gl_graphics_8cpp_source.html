112SEEN_BY_CAMERA GlGraphics::isLocationWithinViewingRange(Vector location)
113{
114 // Make sure the image is within our left and right frustum planes
115 // and if it's not, describe how far away it is to the rest of the system...
116
117 // Just use the left and right planes for right now...
118 float distanceFromLeftPlane = ((myFrustumPlanes[LEFT_PLANE].A * location.x)
119 + (myFrustumPlanes[LEFT_PLANE].B * location.y)
120 + (myFrustumPlanes[LEFT_PLANE].C * location.z)
121 + (myFrustumPlanes[LEFT_PLANE].D));
122
123 float distanceFromRightPlane = ((myFrustumPlanes[RIGHT_PLANE].A * location.x)
124 + (myFrustumPlanes[RIGHT_PLANE].B * location.y)
125 + (myFrustumPlanes[RIGHT_PLANE].C * location.z)
126 + (myFrustumPlanes[RIGHT_PLANE].D));
127
128 // Check to see if an object is outside of our left or right planes. If so, then we don't draw it...
129 if ((distanceFromLeftPlane < -22) || (distanceFromRightPlane < -22))
130 {
131 return SEEN_BY_CAMERA_NO;
132 }
133
134 // Check the distance of an object so we can determine what level of detail should be used ...
135 float distanceFromNearPlane = ((myFrustumPlanes[NEAR_PLANE].A * location.x)
136 + (myFrustumPlanes[NEAR_PLANE].B * location.y)
137 + (myFrustumPlanes[NEAR_PLANE].C * location.z)
138 + (myFrustumPlanes[NEAR_PLANE].D));
139
140 // Check to see how far away an object is. If it's too far, we don't draw it...
141 if (distanceFromNearPlane < -22)
142 {
143 return SEEN_BY_CAMERA_NO;
144 }
145
146 // Ok, we know it's within our viewing frustrum, but just how far away is the sucka?
147 if (distanceFromNearPlane <= 152)
148 {
149 return SEEN_BY_CAMERA_NEAR;
150 }
151 else if ((distanceFromNearPlane > 152) && (distanceFromNearPlane <= 256))
152 {
153 return SEEN_BY_CAMERA_MEDIUM;
154 }
155 else if ((distanceFromNearPlane > 256) && (distanceFromNearPlane <= 384))
156 {
157 return SEEN_BY_CAMERA_FAR;
158 }
159
160 return SEEN_BY_CAMERA_NO;
161}
