279void SpriteEffect::renderSprite(IGraphics *graphics)
280{
281 if(mySpriteInfo.spriteRotationValue.value != 0.0f)
282 {
283 graphics->rotateNextObjectInScene(&mySpriteInfo.spriteRotationValue);
284 }
285
286 if(mySpriteInfo.animationState.currFrame == mySpriteInfo.animationState.endFrame)
287 {
288 mySpriteInfo.Dead = true;
289 }
290
291 // Handle animation here ..
292 mySpriteInfo.sprite3dModel->handleAnimation(gTimeScale, mySpriteInfo.animationState);
293
294 spriteMethodsRenderSprite(&mySpriteInfo, graphics, myOffsetValue, true);
295}
